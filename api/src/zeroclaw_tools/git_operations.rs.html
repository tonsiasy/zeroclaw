23    /// Sanitize git arguments to prevent injection attacks
24    fn sanitize_git_args(&self, args: &str) -> anyhow::Result<Vec<String>> {
25        let mut result = Vec::new();
26        for arg in args.split_whitespace() {
27            // Block dangerous git options that could lead to command injection
28            let arg_lower = arg.to_lowercase();
29            if arg_lower.starts_with("--exec=")
30                || arg_lower.starts_with("--upload-pack=")
31                || arg_lower.starts_with("--receive-pack=")
32                || arg_lower.starts_with("--pager=")
33                || arg_lower.starts_with("--editor=")
34                || arg_lower == "--no-verify"
35                || arg_lower.contains("$(")
36                || arg_lower.contains('`')
37                || arg.contains('|')
38                || arg.contains(';')
39                || arg.contains('>')
40            {
41                anyhow::bail!("Blocked potentially dangerous git argument: {arg}");
42            }
43            // Block `-c` config injection (exact match or `-c=...` prefix).
44            // This must not false-positive on `--cached` or `-cached`.
45            if arg_lower == "-c" || arg_lower.starts_with("-c=") {
46                anyhow::bail!("Blocked potentially dangerous git argument: {arg}");
47            }
48            result.push(arg.to_string());
49        }
50        Ok(result)
51    }

70    /// Resolve a user-provided path to an absolute path within the workspace.
71    /// Returns the workspace_dir if no path is provided.
72    /// Rejects paths that escape the workspace via traversal.
73    fn resolve_working_dir(&self, path: Option<&str>) -> anyhow::Result<std::path::PathBuf> {
74        let base = match path {
75            Some(p) if !p.is_empty() => {
76                let candidate = if std::path::Path::new(p).is_absolute() {
77                    std::path::PathBuf::from(p)
78                } else {
79                    self.workspace_dir.join(p)
80                };
81                let resolved = candidate
82                    .canonicalize()
83                    .map_err(|e| anyhow::anyhow!("Cannot resolve path '{}': {}", p, e))?;
84                let workspace_canonical = self
85                    .workspace_dir
86                    .canonicalize()
87                    .unwrap_or_else(|_| self.workspace_dir.clone());
88                if !resolved.starts_with(&workspace_canonical) {
89                    anyhow::bail!("Path '{}' resolves outside the workspace directory", p);
90                }
91                resolved
92            }
93            _ => self.workspace_dir.clone(),
94        };
95        Ok(base)
96    }

98    async fn run_git_command(
99        &self,
100        args: &[&str],
101        working_dir: &std::path::Path,
102    ) -> anyhow::Result<String> {
103        let output = tokio::process::Command::new("git")
104            .args(args)
105            .current_dir(working_dir)
106            .output()
107            .await?;
108
109        if !output.status.success() {
110            let stderr = String::from_utf8_lossy(&output.stderr);
111            anyhow::bail!("Git command failed: {stderr}");
112        }
113
114        Ok(String::from_utf8_lossy(&output.stdout).to_string())
115    }

117    async fn git_status(
118        &self,
119        _args: serde_json::Value,
120        working_dir: &std::path::Path,
121    ) -> anyhow::Result<ToolResult> {
122        let output = self
123            .run_git_command(&["status", "--porcelain=2", "--branch"], working_dir)
124            .await?;
125
126        // Parse git status output into structured format
127        let mut result = serde_json::Map::new();
128        let mut branch = String::new();
129        let mut staged = Vec::new();
130        let mut unstaged = Vec::new();
131        let mut untracked = Vec::new();
132
133        for line in output.lines() {
134            if line.starts_with("# branch.head ") {
135                branch = line.trim_start_matches("# branch.head ").to_string();
136            } else if let Some(rest) = line.strip_prefix("1 ") {
137                // Ordinary changed entry
138                let mut parts = rest.splitn(3, ' ');
139                if let (Some(staging), Some(path)) = (parts.next(), parts.next())
140                    && !staging.is_empty()
141                {
142                    let status_char = staging.chars().next().unwrap_or(' ');
143                    if status_char != '.' && status_char != ' ' {
144                        staged.push(json!({"path": path, "status": status_char}));
145                    }
146                    let status_char = staging.chars().nth(1).unwrap_or(' ');
147                    if status_char != '.' && status_char != ' ' {
148                        unstaged.push(json!({"path": path, "status": status_char}));
149                    }
150                }
151            } else if let Some(rest) = line.strip_prefix("? ") {
152                untracked.push(rest.to_string());
153            }
154        }
155
156        result.insert("branch".to_string(), json!(branch));
157        result.insert("staged".to_string(), json!(staged));
158        result.insert("unstaged".to_string(), json!(unstaged));
159        result.insert("untracked".to_string(), json!(untracked));
160        result.insert(
161            "clean".to_string(),
162            json!(staged.is_empty() && unstaged.is_empty() && untracked.is_empty()),
163        );
164
165        Ok(ToolResult {
166            success: true,
167            output: serde_json::to_string_pretty(&result).unwrap_or_default(),
168            error: None,
169        })
170    }

172    async fn git_diff(
173        &self,
174        args: serde_json::Value,
175        working_dir: &std::path::Path,
176    ) -> anyhow::Result<ToolResult> {
177        let files = args.get("files").and_then(|v| v.as_str()).unwrap_or(".");
178        let cached = args
179            .get("cached")
180            .and_then(|v| v.as_bool())
181            .unwrap_or(false);
182
183        // Validate files argument against injection patterns
184        self.sanitize_git_args(files)?;
185
186        let mut git_args = vec!["diff", "--unified=3"];
187        if cached {
188            git_args.push("--cached");
189        }
190        git_args.push("--");
191        git_args.push(files);
192
193        let output = self.run_git_command(&git_args, working_dir).await?;
194
195        // Parse diff into structured hunks
196        let mut result = serde_json::Map::new();
197        let mut hunks = Vec::new();
198        let mut current_file = String::new();
199        let mut current_hunk = serde_json::Map::new();
200        let mut lines = Vec::new();
201
202        for line in output.lines() {
203            if line.starts_with("diff --git ") {
204                if !lines.is_empty() {
205                    current_hunk.insert("lines".to_string(), json!(lines));
206                    if !current_hunk.is_empty() {
207                        hunks.push(serde_json::Value::Object(current_hunk.clone()));
208                    }
209                    lines = Vec::new();
210                    current_hunk = serde_json::Map::new();
211                }
212                let parts: Vec<&str> = line.split_whitespace().collect();
213                if parts.len() >= 4 {
214                    current_file = parts[3].trim_start_matches("b/").to_string();
215                    current_hunk.insert("file".to_string(), json!(current_file));
216                }
217            } else if line.starts_with("@@ ") {
218                if !lines.is_empty() {
219                    current_hunk.insert("lines".to_string(), json!(lines));
220                    if !current_hunk.is_empty() {
221                        hunks.push(serde_json::Value::Object(current_hunk.clone()));
222                    }
223                    lines = Vec::new();
224                    current_hunk = serde_json::Map::new();
225                    current_hunk.insert("file".to_string(), json!(current_file));
226                }
227                current_hunk.insert("header".to_string(), json!(line));
228            } else if !line.is_empty() {
229                lines.push(json!({
230                    "text": line,
231                    "type": if line.starts_with('+') { "add" }
232                           else if line.starts_with('-') { "delete" }
233                           else { "context" }
234                }));
235            }
236        }
237
238        if !lines.is_empty() {
239            current_hunk.insert("lines".to_string(), json!(lines));
240            if !current_hunk.is_empty() {
241                hunks.push(serde_json::Value::Object(current_hunk));
242            }
243        }
244
245        result.insert("hunks".to_string(), json!(hunks));
246        result.insert("file_count".to_string(), json!(hunks.len()));
247
248        Ok(ToolResult {
249            success: true,
250            output: serde_json::to_string_pretty(&result).unwrap_or_default(),
251            error: None,
252        })
253    }

255    async fn git_log(
256        &self,
257        args: serde_json::Value,
258        working_dir: &std::path::Path,
259    ) -> anyhow::Result<ToolResult> {
260        let limit_raw = args.get("limit").and_then(|v| v.as_u64()).unwrap_or(10);
261        let limit = usize::try_from(limit_raw).unwrap_or(usize::MAX).min(1000);
262        let limit_str = limit.to_string();
263
264        let output = self
265            .run_git_command(
266                &[
267                    "log",
268                    &format!("-{limit_str}"),
269                    "--pretty=format:%H|%an|%ae|%ad|%s",
270                    "--date=iso",
271                ],
272                working_dir,
273            )
274            .await?;
275
276        let mut commits = Vec::new();
277
278        for line in output.lines() {
279            let parts: Vec<&str> = line.split('|').collect();
280            if parts.len() >= 5 {
281                commits.push(json!({
282                    "hash": parts[0],
283                    "author": parts[1],
284                    "email": parts[2],
285                    "date": parts[3],
286                    "message": parts[4]
287                }));
288            }
289        }
290
291        Ok(ToolResult {
292            success: true,
293            output: serde_json::to_string_pretty(&json!({ "commits": commits }))
294                .unwrap_or_default(),
295            error: None,
296        })
297    }

299    async fn git_branch(
300        &self,
301        _args: serde_json::Value,
302        working_dir: &std::path::Path,
303    ) -> anyhow::Result<ToolResult> {
304        let output = self
305            .run_git_command(
306                &["branch", "--format=%(refname:short)|%(HEAD)"],
307                working_dir,
308            )
309            .await?;
310
311        let mut branches = Vec::new();
312        let mut current = String::new();
313
314        for line in output.lines() {
315            if let Some((name, head)) = line.split_once('|') {
316                let is_current = head == "*";
317                if is_current {
318                    current = name.to_string();
319                }
320                branches.push(json!({
321                    "name": name,
322                    "current": is_current
323                }));
324            }
325        }
326
327        Ok(ToolResult {
328            success: true,
329            output: serde_json::to_string_pretty(&json!({
330                "current": current,
331                "branches": branches
332            }))
333            .unwrap_or_default(),
334            error: None,
335        })
336    }

339        if message.chars().count() > 2000 {

341        } else {

346    async fn git_commit(
347        &self,
348        args: serde_json::Value,
349        working_dir: &std::path::Path,
350    ) -> anyhow::Result<ToolResult> {
351        let message = args
352            .get("message")
353            .and_then(|v| v.as_str())
354            .ok_or_else(|| anyhow::anyhow!("Missing 'message' parameter"))?;
355
356        // Sanitize commit message
357        let sanitized = message
358            .lines()
359            .map(|l| l.trim())
360            .filter(|l| !l.is_empty())
361            .collect::<Vec<_>>()
362            .join("\n");
363
364        if sanitized.is_empty() {
365            anyhow::bail!("Commit message cannot be empty");
366        }
367
368        // Limit message length
369        let message = Self::truncate_commit_message(&sanitized);
370
371        let output = self
372            .run_git_command(&["commit", "-m", &message], working_dir)
373            .await;
374
375        match output {
376            Ok(_) => Ok(ToolResult {
377                success: true,
378                output: format!("Committed: {message}"),
379                error: None,
380            }),
381            Err(e) => Ok(ToolResult {
382                success: false,
383                output: String::new(),
384                error: Some(format!("Commit failed: {e}")),
385            }),
386        }
387    }

420    async fn git_checkout(
421        &self,
422        args: serde_json::Value,
423        working_dir: &std::path::Path,
424    ) -> anyhow::Result<ToolResult> {
425        let branch = args
426            .get("branch")
427            .and_then(|v| v.as_str())
428            .ok_or_else(|| anyhow::anyhow!("Missing 'branch' parameter"))?;
429
430        // Sanitize branch name
431        let sanitized = self.sanitize_git_args(branch)?;
432
433        if sanitized.is_empty() || sanitized.len() > 1 {
434            anyhow::bail!("Invalid branch specification");
435        }
436
437        let branch_name = &sanitized[0];
438
439        // Block dangerous branch names
440        if branch_name.contains('@') || branch_name.contains('^') || branch_name.contains('~') {
441            anyhow::bail!("Branch name contains invalid characters");
442        }
443
444        let output = self
445            .run_git_command(&["checkout", branch_name], working_dir)
446            .await;
447
448        match output {
449            Ok(_) => Ok(ToolResult {
450                success: true,
451                output: format!("Switched to branch: {branch_name}"),
452                error: None,
453            }),
454            Err(e) => Ok(ToolResult {
455                success: false,
456                output: String::new(),
457                error: Some(format!("Checkout failed: {e}")),
458            }),

462    async fn git_stash(
463        &self,
464        args: serde_json::Value,
465        working_dir: &std::path::Path,
466    ) -> anyhow::Result<ToolResult> {
467        let action = args
468            .get("action")
469            .and_then(|v| v.as_str())
470            .unwrap_or("push");
471
472        let output = match action {
473            "push" | "save" => {
474                self.run_git_command(&["stash", "push", "-m", "auto-stash"], working_dir)
475                    .await
476            }
477            "pop" => self.run_git_command(&["stash", "pop"], working_dir).await,
478            "list" => self.run_git_command(&["stash", "list"], working_dir).await,
479            "drop" => {
480                let index_raw = args.get("index").and_then(|v| v.as_u64()).unwrap_or(0);
481                let index = i32::try_from(index_raw)
482                    .map_err(|_| anyhow::anyhow!("stash index too large: {index_raw}"))?;
483                self.run_git_command(
484                    &["stash", "drop", &format!("stash@{{{index}}}")],
485                    working_dir,
486                )
487                .await
488            }
489            _ => anyhow::bail!("Unknown stash action: {action}. Use: push, pop, list, drop"),
490        };
491
492        match output {
493            Ok(out) => Ok(ToolResult {
494                success: true,
495                output: out,
496                error: None,
497            }),
498            Err(e) => Ok(ToolResult {
499                success: false,
500                output: String::new(),
501                error: Some(format!("Stash {action} failed: {e}")),
502            }),
503        }
504    }

558                },

562                }
563            },
564            "required": ["operation"]
565        })
566    }
567
568    async fn execute(&self, args: serde_json::Value) -> anyhow::Result<ToolResult> {
569        let operation = match args.get("operation").and_then(|v| v.as_str()) {
570            Some(op) => op,
571            None => {
572                return Ok(ToolResult {
573                    success: false,
574                    output: String::new(),
575                    error: Some("Missing 'operation' parameter".into()),
576                });
577            }
578        };
579
580        let path = args.get("path").and_then(|v| v.as_str());
581        let working_dir = match self.resolve_working_dir(path) {
582            Ok(d) => d,
583            Err(e) => {
584                return Ok(ToolResult {
585                    success: false,
586                    output: String::new(),
587                    error: Some(format!("Invalid path: {e}")),
588                });
589            }
590        };
591
592        // Check if we're in a git repository
593        if !working_dir.join(".git").exists() {
594            // Try to find .git in parent directories
595            let mut current_dir = working_dir.as_path();
596            let mut found_git = false;
597            while current_dir.parent().is_some() {
598                if current_dir.join(".git").exists() {
599                    found_git = true;
600                    break;
601                }
602                current_dir = current_dir.parent().unwrap();
603            }
604
605            if !found_git {
606                return Ok(ToolResult {
607                    success: false,
608                    output: String::new(),
609                    error: Some("Not in a git repository".into()),
610                });
611            }
612        }
613
614        // Check autonomy level for write operations
615        if self.requires_write_access(operation) {
616            if !self.security.can_act() {
617                return Ok(ToolResult {
618                    success: false,
619                    output: String::new(),
620                    error: Some(
621                        "Action blocked: git write operations require higher autonomy level".into(),
622                    ),
623                });
624            }
625
626            match self.security.autonomy {
627                AutonomyLevel::ReadOnly => {
628                    return Ok(ToolResult {
629                        success: false,
630                        output: String::new(),
631                        error: Some("Action blocked: read-only mode".into()),
632                    });
633                }
634                AutonomyLevel::Supervised | AutonomyLevel::Full => {}
635            }
636        }
637
638        // Record action for rate limiting
639        if !self.security.record_action() {
640            return Ok(ToolResult {
641                success: false,
642                output: String::new(),
643                error: Some("Action blocked: rate limit exceeded".into()),
644            });
645        }
646
647        // Execute the requested operation
648        match operation {
649            "status" => self.git_status(args, &working_dir).await,
650            "diff" => self.git_diff(args, &working_dir).await,
651            "log" => self.git_log(args, &working_dir).await,
652            "branch" => self.git_branch(args, &working_dir).await,
653            "commit" => self.git_commit(args, &working_dir).await,
654            "add" => self.git_add(args, &working_dir).await,
655            "checkout" => self.git_checkout(args, &working_dir).await,
656            "stash" => self.git_stash(args, &working_dir).await,
657            _ => Ok(ToolResult {
658                success: false,
659                output: String::new(),
660                error: Some(format!("Unknown operation: {operation}")),
661            }),
662        }
663    }
