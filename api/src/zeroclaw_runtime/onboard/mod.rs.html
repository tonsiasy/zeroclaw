1//! Onboard orchestrator.
2//!
3//! Thin dispatcher above the `OnboardUi` trait (defined in
4//! `zeroclaw-config::traits`). Section-scoped entry points let callers run
5//! just one slice (`zeroclaw onboard channels`) or the whole flow.
6//!
7//! Everything writes through `Config::set_prop` (or its helpers); direct
8//! struct-field assignment is off-limits per the DRY contract (#5951).
9
10use anyhow::Result;
11use zeroclaw_config::schema::Config;
12use zeroclaw_config::traits::{Answer, OnboardUi, PropKind, SelectItem};
13
14/// Internal prompt / section navigation signal. `Done` = advance. `Back` =
15/// the user pressed Esc; rewind one step. Helpers propagate it up through
16/// `prompt_field` → `prompt_fields_under` → section fn → `run_all`.
17#[derive(Debug, Clone, Copy, PartialEq, Eq)]
18enum Nav {
19    Done,
20    Back,
21}
22
23/// Skip-gate outcome. `Skip` = section already configured, user chose not
24/// to reconfigure. `Enter` = walk the section. `Back` = user pressed Esc
25/// at the skip prompt, bounce to the previous section.
26#[derive(Debug, Clone, Copy, PartialEq, Eq)]
27enum SkipNav {
28    Skip,
29    Enter,
30    Back,
31}

33pub mod ui;

35/// Which slice of onboarding to run. `All` runs every section in order.
36#[derive(Debug, Clone, Copy, PartialEq, Eq)]
37pub enum Section {
38    All,
39    Workspace,
40    Providers,
41    Channels,
42    Memory,
43    Hardware,
44    Tunnel,
45}
46
47/// Runtime knobs sourced from CLI flags. `--quick`/`--tui` select the UI
48/// backend at the binary edge and don't appear here — the orchestrator only
49/// cares about per-section behavior.
50#[derive(Debug, Default, Clone)]
51pub struct Flags {
52    /// Skip "keep existing value?" confirmations; always re-prompt.
53    pub force: bool,
54    /// Back up the current config dir and start from `Config::default()`.
55    pub reinit: bool,
56    pub api_key: Option<String>,
57    pub provider: Option<String>,
58    pub model: Option<String>,
59    pub memory: Option<String>,
60}

62/// Top-level onboard dispatcher.
63pub async fn run(
64    cfg: &mut Config,
65    ui: &mut dyn OnboardUi,
66    section: Section,
67    flags: &Flags,
68) -> Result<()> {
69    match section {
70        Section::All => run_all(cfg, ui, flags).await,
71        Section::Workspace => {
72            let _ = workspace(cfg, ui, flags).await?;
73            Ok(())
74        }
75        Section::Providers => {
76            let _ = providers(cfg, ui, flags).await?;
77            Ok(())
78        }
79        Section::Channels => {
80            let _ = channels(cfg, ui, flags).await?;
81            Ok(())
82        }
83        Section::Memory => {
84            let _ = memory(cfg, ui, flags).await?;
85            Ok(())
86        }
87        Section::Hardware => {
88            let _ = hardware(cfg, ui, flags).await?;
89            Ok(())
90        }
91        Section::Tunnel => {
92            let _ = tunnel(cfg, ui, flags).await?;
93            Ok(())
94        }
95    }
96}

98/// Walk every section in order with section-level Back. Each section returns
99/// `Nav::Back` when the user pressed Esc at its first prompt; the loop
100/// rewinds to the previous section. Back at the first section exits
101/// onboarding cleanly (user bails out).
102async fn run_all(cfg: &mut Config, ui: &mut dyn OnboardUi, flags: &Flags) -> Result<()> {
103    let mut i: usize = 0;
104    loop {
105        let nav = match i {
106            0 => workspace(cfg, ui, flags).await?,
107            1 => providers(cfg, ui, flags).await?,
108            2 => channels(cfg, ui, flags).await?,
109            3 => memory(cfg, ui, flags).await?,
110            4 => hardware(cfg, ui, flags).await?,
111            5 => tunnel(cfg, ui, flags).await?,
112            _ => return Ok(()),
113        };
114        match nav {
115            Nav::Done => i += 1,
116            Nav::Back => {
117                if i == 0 {
118                    return Ok(());
119                }
120                i -= 1;
121            }
122        }
123    }
124}

126/// Write a single property and immediately persist the whole config. This is
127/// the ONE path every section takes to mutate cfg, so users who Ctrl+C
128/// mid-flow find their prior answers already saved on disk — re-running
129/// `zeroclaw onboard` picks up where they left off.
130async fn persist(cfg: &mut Config, path: &str, value: &str) -> Result<()> {
131    cfg.set_prop(path, value)?;
132    cfg.save().await?;
133    Ok(())
134}

138/// Per-field default override. When a section knows a sensible default
139/// that lives outside the config (e.g. `AnthropicProvider::default_temperature()`),
140/// it builds a list of these and passes them to `prompt_fields_under`.
141/// The prompt surfaces the default — shown in the label as
142/// `"timeout-secs (default: 120)"` and prefilled into the input so the
143/// user just hits Enter to accept — only when the field is unset in cfg.
144#[derive(Debug, Clone)]
145pub struct FieldDefault {
146    pub path: String,
147    pub display: String,
148}
149
150fn find_default<'a>(defaults: &'a [FieldDefault], path: &str) -> Option<&'a str> {
151    defaults
152        .iter()
153        .find(|d| d.path == path)
154        .map(|d| d.display.as_str())
155}

157/// True when `input` parses as the same `Vec<String>` form `config.toml`
158/// emits. Lets the StringArray prompt accept the bracketed display form
159/// bidirectionally.
160fn parses_as_string_array(input: &str) -> bool {
161    toml::from_str::<std::collections::HashMap<String, Vec<String>>>(&format!("v = {input}"))
162        .is_ok()
163}

165/// Prompt for a single config field identified by its dotted name. Returns
166/// `Nav::Back` when the user pressed Esc at the prompt; `Nav::Done` on any
167/// other outcome (including "kept current value"). `default` is the
168/// section-supplied fallback for unset fields — surfaced in the label and
169/// prefilled into the input.
170async fn prompt_field(
171    cfg: &mut Config,
172    ui: &mut dyn OnboardUi,
173    name: &str,
174    default: Option<&str>,
175) -> Result<Nav> {
176    let field = cfg
177        .prop_fields()
178        .into_iter()
179        .find(|f| f.name == name)
180        .ok_or_else(|| anyhow::anyhow!("unknown config field: {name}"))?;
181
182    let short = name.rsplit('.').next().unwrap_or(name);
183    let current = field.display_value;
184    // For bools, `display_value` is always `"true"` or `"false"` — never
185    // empty, never `"<unset>"` — so a naive is-set check can't tell an
186    // explicit user choice apart from the struct-level default. Treat
187    // bools as unset here: the [Yes]/[No] toggle already surfaces the
188    // current state, and collapsing `is_set` lets any passed `default`
189    // render in the prompt label (`enabled (default: true)`) while
190    // keeping the misleading "Current: …" annotation out of the help.
191    let is_set = field.kind != PropKind::Bool && !current.is_empty() && current != "<unset>";
192
193    // Surface the docstring as help text above the prompt, and append
194    // whichever annotation fits the prompt's state: "Default: X" when
195    // the section supplied one and the field is unset, "Current: X"
196    // when the config carries a user-set value (non-bool only).
197    let mut help = field.description.to_string();
198    // List-of-strings fields take comma-separated input. Without this
199    // hint users guess and end up entering things like `["alice"]` as
200    // raw text — the parser then treats that as one big string element
201    // and the saved config is garbage.
202    if field.kind == PropKind::StringArray {
203        if !help.is_empty() {
204            help.push('\n');
205        }
206        help.push_str("Format: alice,bob or [\"alice\", \"bob\"]. Empty = clear list.");
207    }
208    if !is_set
209        && let Some(d) = default
210        && !d.is_empty()
211    {
212        if !help.is_empty() {
213            help.push('\n');
214        }
215        help.push_str(&format!("Default: {d}. Press Enter to accept."));
216    } else if is_set {
217        if !help.is_empty() {
218            help.push('\n');
219        }
220        help.push_str(&format!("Current: {current}. Enter to keep."));
221    }
222    ui.note(&help);
223
224    // Label decorates the short name with the default (when visible) so the
225    // value is anchored to the prompt line itself, not just the help text.
226    let prompt_label = match (is_set, default) {
227        (false, Some(d)) if !d.is_empty() => format!("{short} (default: {d})"),
228        _ => short.to_string(),
229    };
230    let prompt = prompt_label.as_str();
231
232    if field.is_secret {
233        match ui.secret(prompt, is_set).await? {
234            Answer::Back => return Ok(Nav::Back),
235            Answer::Value(Some(value)) => persist(cfg, name, &value).await?,
236            Answer::Value(None) => {}
237        }
238        return Ok(Nav::Done);
239    }
240
241    match field.kind {
242        PropKind::Bool => {
243            let cur = current.parse::<bool>().unwrap_or(false);
244            match ui.confirm(prompt, cur).await? {
245                Answer::Back => return Ok(Nav::Back),
246                Answer::Value(new) if new != cur => persist(cfg, name, &new.to_string()).await?,
247                Answer::Value(_) => {}
248            }
249        }
250        PropKind::String | PropKind::Integer | PropKind::Float => {
251            // Prefill priority: config current value > section default > empty.
252            // When the user accepts the prefilled default (no edit), we
253            // still write it through set_prop so the config records the
254            // resolved value rather than leaving it as an implicit fallback.
255            let prefill = if is_set {
256                Some(current.as_str())
257            } else {
258                default
259            };
260            match ui.string(prompt, prefill).await? {
261                Answer::Back => return Ok(Nav::Back),
262                Answer::Value(new) => {
263                    if (is_set || !new.is_empty()) && new != current {
264                        persist(cfg, name, &new).await?;
265                    }
266                }
267            }
268        }
269        PropKind::StringArray => {
270            let prefill = if is_set {
271                Some(current.as_str())
272            } else {
273                default
274            };
275            // Accepts comma-separated input or the bracketed form from
276            // config.toml. Reject malformed brackets — otherwise the
277            // parser silently coerces them into a single-element list
278            // of garbage.
279            loop {
280                match ui.string(prompt, prefill).await? {
281                    Answer::Back => return Ok(Nav::Back),
282                    Answer::Value(new) => {
283                        let trimmed = new.trim();
284                        if trimmed.starts_with('[') && !parses_as_string_array(trimmed) {
285                            ui.note("Invalid array. Use alice,bob or [\"alice\", \"bob\"].");
286                            continue;
287                        }
288                        if (is_set || !new.is_empty()) && new != current {
289                            persist(cfg, name, &new).await?;
290                        }
291                        ui.note("");
292                        break;
293                    }
294                }
295            }
296        }
297        PropKind::Enum => {
298            let variants = field.enum_variants.map(|get| get()).unwrap_or_default();
299            if variants.is_empty() {
300                ui.warn(&format!("skipping {name}: no enum variants exposed"));
301                return Ok(Nav::Done);
302            }
303            let items: Vec<SelectItem> = variants.iter().map(SelectItem::new).collect();
304            let current_idx = if is_set {
305                variants.iter().position(|v| v == &current)
306            } else {
307                default.and_then(|d| variants.iter().position(|v| v == d))
308            };
309            match ui.select(prompt, &items, current_idx).await? {
310                Answer::Back => return Ok(Nav::Back),
311                Answer::Value(idx) => {
312                    let new = &variants[idx];
313                    if new != &current {
314                        persist(cfg, name, new).await?;
315                    }
316                }
317            }
318        }
319    }
320    Ok(Nav::Done)
321}

323/// Iterate every field under `prefix` in `prop_fields()` and prompt for each.
324/// `excludes` lists leaf field names to skip. `defaults` carries per-field
325/// fallback values (e.g. provider-trait defaults) surfaced in the prompt
326/// when the field is unset. Rewinds on `Nav::Back`; propagates `Back` to
327/// the caller when the user rewinds past the first prompt.
328async fn prompt_fields_under(
329    cfg: &mut Config,
330    ui: &mut dyn OnboardUi,
331    prefix: &str,
332    excludes: &[&str],
333    defaults: &[FieldDefault],
334) -> Result<Nav> {
335    let names: Vec<String> = cfg
336        .prop_fields()
337        .into_iter()
338        .filter_map(|f| {
339            let suffix = f.name.strip_prefix(prefix)?.strip_prefix('.')?;
340            if suffix.contains('.') || excludes.contains(&suffix) {
341                return None;
342            }
343            Some(f.name.to_string())
344        })
345        .collect();
346    let mut i: usize = 0;
347    while i < names.len() {
348        let default = find_default(defaults, &names[i]);
349        match prompt_field(cfg, ui, &names[i], default).await? {
350            Nav::Done => i += 1,
351            Nav::Back => {
352                if i == 0 {
353                    return Ok(Nav::Back);
354                }
355                i -= 1;
356            }
357        }
358    }
359    Ok(Nav::Done)
360}

362/// Section-level skip gate. A section is "already configured" when EITHER
363/// (a) it has a marker in `onboard_state.completed_sections` (user finished
364/// the flow once), OR (b) the caller supplies a section-specific
365/// has-meaningful-config signal (e.g. workspace.enabled == true, providers
366/// has a fallback + api-key set). `--force` bypasses unconditionally.
367async fn skip_if_configured(
368    cfg: &Config,
369    ui: &mut dyn OnboardUi,
370    flags: &Flags,
371    section_key: &str,
372    label: &str,
373    has_signal: bool,
374) -> Result<SkipNav> {
375    if flags.force {
376        return Ok(SkipNav::Enter);
377    }
378    let seen = cfg
379        .onboard_state
380        .completed_sections
381        .iter()
382        .any(|s| s == section_key);
383    if !seen && !has_signal {
384        return Ok(SkipNav::Enter);
385    }
386    match ui
387        .confirm(
388            &format!("{label} is already configured. Reconfigure?"),
389            false,
390        )
391        .await?
392    {
393        Answer::Back => Ok(SkipNav::Back),
394        Answer::Value(true) => Ok(SkipNav::Enter),
395        Answer::Value(false) => Ok(SkipNav::Skip),
396    }
397}

399/// Per-section meaningful-config detector used as the secondary skip-gate
400/// signal alongside the completed_sections marker. Returns true when the
401/// section has values that can only come from user action (i.e. diverged
402/// from `Config::default()`'s idle state).
403fn section_has_signal(cfg: &Config, section_key: &str) -> bool {
404    match section_key {
405        "workspace" => cfg.workspace.enabled,
406        "providers" => cfg.providers.fallback.is_some() && !cfg.providers.models.is_empty(),
407        // `channels.cli: bool` is a default-true scalar that lives directly
408        // under `channels.*`, so a bare `starts_with("channels.")` check
409        // fires on every fresh install. Require a nested channel config
410        // (e.g. `channels.telegram.bot-token`) — anything with a second dot
411        // segment — to count as user-driven signal.
412        "channels" => cfg.prop_fields().iter().any(|f| {
413            f.name
414                .strip_prefix("channels.")
415                .is_some_and(|rest| rest.contains('.'))
416        }),
417        "hardware" => cfg.hardware.enabled,
418        // Memory's default backend is "sqlite" and Tunnel's is "none" — both
419        // are valid user choices indistinguishable from untouched defaults.
420        // Marker-only for these two.
421        _ => false,
422    }
423}

425/// Record that a section finished so the next run's skip gate can fire.
426async fn mark_completed(cfg: &mut Config, section_key: &str) -> Result<()> {
427    if cfg
428        .onboard_state
429        .completed_sections
430        .iter()
431        .any(|s| s == section_key)
432    {
433        return Ok(());
434    }
435    cfg.onboard_state
436        .completed_sections
437        .push(section_key.to_string());
438    cfg.save().await?;
439    Ok(())
440}

442// ── Sections ─────────────────────────────────────────────────────────────
443// Each section returns `Nav::Back` when the user hits Esc at the very first
444// prompt. Back from a later prompt within the section rewinds locally (via
445// prompt_fields_under / per-section loop), never propagates to the parent.
446
447async fn workspace(cfg: &mut Config, ui: &mut dyn OnboardUi, flags: &Flags) -> Result<Nav> {
448    ui.heading(1, "Workspace");
449    ui.status(&format!(
450        "Workspace directory: {}",
451        cfg.workspace_dir.display()
452    ));
453    match skip_if_configured(
454        cfg,
455        ui,
456        flags,
457        "workspace",
458        "Workspace",
459        section_has_signal(cfg, "workspace"),
460    )
461    .await?
462    {
463        SkipNav::Skip => return Ok(Nav::Done),
464        SkipNav::Back => return Ok(Nav::Back),
465        SkipNav::Enter => {}
466    }
467
468    loop {
469        match prompt_field(cfg, ui, "workspace.enabled", None).await? {
470            Nav::Back => return Ok(Nav::Back),
471            Nav::Done => {}
472        }
473        if cfg.workspace.enabled {
474            match prompt_fields_under(cfg, ui, "workspace", &["enabled"], &[]).await? {
475                Nav::Back => continue,
476                Nav::Done => break,
477            }
478        } else {
479            break;
480        }
481    }
482
483    mark_completed(cfg, "workspace").await?;
484    Ok(Nav::Done)
485}

487async fn providers(cfg: &mut Config, ui: &mut dyn OnboardUi, flags: &Flags) -> Result<Nav> {
488    ui.heading(1, "Providers");
489    if flags.provider.is_none() && flags.api_key.is_none() && flags.model.is_none() {
490        match skip_if_configured(
491            cfg,
492            ui,
493            flags,
494            "providers",
495            "Providers",
496            section_has_signal(cfg, "providers"),
497        )
498        .await?
499        {
500            SkipNav::Skip => return Ok(Nav::Done),
501            SkipNav::Back => return Ok(Nav::Back),
502            SkipNav::Enter => {}
503        }
504    }
505    // Surface both auth paths up front so users with an existing key go
506    // straight to the api_key prompt, and users on OAuth-only providers
507    // (Codex, Claude Code, etc.) know to use the separate login flow.
508    ui.note(
509        "Paste an API key (e.g. `sk-ant-…` for Anthropic, `sk-…` for OpenAI) \
510         when prompted. For OAuth-based providers run: \
511         zeroclaw auth login --provider <name>",
512    );
513
514    // Menu is driven by zeroclaw_providers::list_providers() — single source
515    // of truth for canonical names, display names, aliases.
516    let entries = zeroclaw_providers::list_providers();
517
518    loop {
519        let current_fallback = cfg.providers.fallback.clone().unwrap_or_default();
520
521        let picked = match &flags.provider {
522            Some(forced) => forced.clone(),
523            None => {
524                let current_idx = entries.iter().position(|p| p.name == current_fallback);
525                let mut options: Vec<SelectItem> = entries
526                    .iter()
527                    .map(|p| {
528                        let configured = cfg.providers.models.contains_key(p.name);
529                        let is_active = p.name == current_fallback;
530                        let badge = match (is_active, configured) {
531                            (true, _) => Some("[active]".into()),
532                            (_, true) => Some("[configured]".into()),
533                            _ => None,
534                        };
535                        SelectItem {
536                            label: p.display_name.to_string(),
537                            badge,
538                        }
539                    })
540                    .collect();
541                // "Done" lets the user exit providers without picking one —
542                // matches the channels picker's escape hatch. Highlight it
543                // by default when no fallback is set yet (first-time setup).
544                let done_idx = options.len();
545                options.push(SelectItem::new("Done"));
546                let initial = current_idx.or(Some(done_idx));
547                let idx = match ui.select("Provider", &options, initial).await? {
548                    Answer::Back => return Ok(Nav::Back),
549                    Answer::Value(idx) => idx,
550                };
551                if idx == done_idx {
552                    break;
553                }
554                entries[idx].name.to_string()
555            }
556        };
557
558        // Seed the HashMap entry in memory so `prop_fields` can enumerate
559        // its fields for the prompts below. Not persisted here — the first
560        // `persist()` for a real value (api_key, model, …) carries it to
561        // disk. If the user backs out before any value is set, the back
562        // paths drop the entry so it never reaches the file.
563        let is_new_entry = !cfg.providers.models.contains_key(&picked);
564        cfg.providers.models.entry(picked.clone()).or_default();
565
566        // For fresh entries, pre-populate the provider's trait-level defaults
567        // into the in-memory entry. Skipped when reconfiguring so existing
568        // user overrides aren't clobbered. Lives in memory until the first
569        // `persist()` carries the entry — defaults included — to disk.
570        if is_new_entry {
571            let prefix = format!("providers.models.{picked}");
572            for trait_default in provider_trait_defaults(&picked, &prefix) {
573                cfg.set_prop(&trait_default.path, &trait_default.display)?;
574            }
575        }
576
577        let display_name = entries
578            .iter()
579            .find(|p| p.name == picked)
580            .map(|p| p.display_name)
581            .unwrap_or(picked.as_str());
582        ui.heading(2, display_name);
583
584        // Apply CLI-flag overrides up front, then skip those names in the
585        // interactive pass so the user isn't re-prompted for what they already
586        // passed on the command line.
587        let prefix = format!("providers.models.{picked}");
588        let api_key_path = format!("{prefix}.api-key");
589        if let Some(api_key) = &flags.api_key {
590            persist(cfg, &api_key_path, api_key).await?;
591        }
592        if let Some(model) = &flags.model {
593            persist(cfg, &format!("{prefix}.model"), model).await?;
594        }
595
596        // Authentication phase is prompted explicitly so the user sees a
597        // clear "API key" step, not a generic `api-key (stored, replace?)`
598        // lost among other fields. The heading(2) also overrides the
599        // provider subsection so the panel reads "Providers › Authentication".
600        if flags.api_key.is_none() {
601            ui.heading(2, &format!("{display_name} › Authentication"));
602            match prompt_field(cfg, ui, &api_key_path, None).await? {
603                Nav::Back => {
604                    if flags.provider.is_some() {
605                        return Ok(Nav::Back);
606                    }
607                    cfg.providers.models.remove(&picked);
608                    continue;
609                }
610                Nav::Done => {}
611            }
612            ui.heading(2, display_name);
613        }
614
615        if flags.model.is_none() {
616            ui.heading(2, &format!("{display_name} › Model"));
617            match prompt_model(cfg, ui, &picked).await? {
618                Nav::Back => {
619                    if flags.provider.is_some() {
620                        return Ok(Nav::Back);
621                    }
622                    cfg.providers.models.remove(&picked);
623                    continue;
624                }
625                Nav::Done => {}
626            }
627            ui.heading(2, display_name);
628        }
629
630        // Advanced settings (temperature, timeout, base-url override,
631        // wire-api, etc.) are gated behind an opt-in. Most users never
632        // touch these, and the trait-level defaults are sensible.
633        match offer_advanced_settings(cfg, ui, &picked, &prefix).await? {
634            Nav::Back => {
635                if flags.provider.is_some() {
636                    return Ok(Nav::Back);
637                }
638                continue;
639            }
640            Nav::Done => {}
641        }
642
643        break;
644    }
645
646    mark_completed(cfg, "providers").await?;
647    Ok(Nav::Done)
648}

650/// Opt-in gate for the per-provider advanced field sweep. Default N so the
651/// user breezes through onboarding after auth + model; Y walks them through
652/// every remaining field (temperature, max_tokens, timeout_secs, base_url,
653/// wire_api, azure_*, etc.) with the provider's trait defaults pre-filled.
654async fn offer_advanced_settings(
655    cfg: &mut Config,
656    ui: &mut dyn OnboardUi,
657    provider: &str,
658    prefix: &str,
659) -> Result<Nav> {
660    ui.heading(2, "Advanced settings");
661    ui.note(
662        "Temperature, timeout, base-URL override, wire protocol, etc. The \
663         provider's own defaults are used when these are left unset — skip \
664         unless you need to override something specific.",
665    );
666    match ui.confirm("Configure advanced settings?", false).await? {
667        Answer::Back => return Ok(Nav::Back),
668        Answer::Value(false) => return Ok(Nav::Done),
669        Answer::Value(true) => {}
670    }
671
672    let defaults = provider_trait_defaults(provider, prefix);
673
674    // Skipped: `model` (already via prompt_model), `api-key` (explicit auth
675    // phase), and fields that only apply to a different provider family
676    // (e.g. azure-openai-* when the user picked Anthropic).
677    let mut excludes: Vec<&str> = vec!["model", "api-key"];
678    excludes.extend(provider_family_excludes(provider));
679    prompt_fields_under(cfg, ui, prefix, &excludes, &defaults).await
680}

682/// Build the `FieldDefault` list for a provider from its trait methods.
683/// Single source of truth: used to pre-populate fresh entries during
684/// onboarding AND to surface defaults in the advanced-settings walk.
685/// A provider handle constructed with `api_key = None` can still answer
686/// the `default_*` questions — they're family-level constants, not
687/// runtime values. If construction fails (unknown / local-only provider),
688/// returns an empty vec and callers fall back to per-field display values.
689fn provider_trait_defaults(provider: &str, prefix: &str) -> Vec<FieldDefault> {
690    let Ok(handle) = zeroclaw_providers::create_provider(provider, None) else {
691        return Vec::new();
692    };
693    let mut v = vec![
694        FieldDefault {
695            path: format!("{prefix}.temperature"),
696            display: handle.default_temperature().to_string(),
697        },
698        FieldDefault {
699            path: format!("{prefix}.max-tokens"),
700            display: handle.default_max_tokens().to_string(),
701        },
702        FieldDefault {
703            path: format!("{prefix}.timeout-secs"),
704            display: handle.default_timeout_secs().to_string(),
705        },
706        FieldDefault {
707            path: format!("{prefix}.wire-api"),
708            display: handle.default_wire_api().to_string(),
709        },
710    ];
711    if let Some(url) = handle.default_base_url() {
712        v.push(FieldDefault {
713            path: format!("{prefix}.base-url"),
714            display: url.to_string(),
715        });
716    }
717    v
718}

720/// Exclude fields that don't apply to the selected provider family.
721///
722/// A field absent from this filter is shown for every provider. Grow the
723/// list as provider-specific fields land on `ModelProviderConfig` — the
724/// goal is to keep the advanced walk focused on fields the user might
725/// reasonably want to override for the provider they picked.
726///
727/// Follow-up (#5960 deferral): a `#[configurable(family = "...")]`
728/// attribute on the schema struct would let the `Configurable` derive
729/// emit this filter automatically. This hardcoded list is a pragmatic
730/// stand-in until that lands.
731fn provider_family_excludes(provider: &str) -> Vec<&'static str> {
732    let mut out = Vec::new();
733    // Azure-OpenAI-only deployment routing fields.
734    if provider != "azure_openai" {
735        out.push("azure-openai-resource");
736        out.push("azure-openai-deployment");
737        out.push("azure-openai-api-version");
738    }
739    // OpenAI wire-protocol flavor + Codex auth are OpenAI-family concerns.
740    if !matches!(provider, "openai" | "openai_codex") {
741        out.push("wire-api");
742        out.push("requires-openai-auth");
743    }
744    out
745}

747/// Prompt for the model field using the provider's live model catalog.
748///
749/// Calls `Provider::list_models()` (no auth — see `zeroclaw-providers`
750/// models_dev + native public endpoints). Falls back to a manual string
751/// input when the provider doesn't expose a no-auth list or the fetch fails.
752async fn prompt_model(cfg: &mut Config, ui: &mut dyn OnboardUi, provider: &str) -> Result<Nav> {
753    let model_path = format!("providers.models.{provider}.model");
754    let current = cfg.get_prop(&model_path).unwrap_or_default();
755    let is_set = !current.is_empty() && current != "<unset>";
756
757    let live_models = match zeroclaw_providers::create_provider(provider, None) {
758        Ok(handle) => {
759            ui.status("Fetching models...");
760            match handle.list_models().await {
761                Ok(models) => Some(models),
762                Err(e) => {
763                    tracing::debug!(provider, error = ?e, "models.dev catalog fetch failed");
764                    None
765                }
766            }
767        }
768        Err(e) => {
769            tracing::debug!(
770                provider,
771                error = ?e,
772                "provider construction failed for model-list probe"
773            );
774            None
775        }
776    };
777
778    let new_value = match live_models.filter(|ms| !ms.is_empty()) {
779        Some(models) => {
780            let items: Vec<SelectItem> = models.iter().map(SelectItem::new).collect();
781            let current_idx = models.iter().position(|m| m == &current);
782            match ui.select("Model", &items, current_idx).await? {
783                Answer::Back => return Ok(Nav::Back),
784                Answer::Value(idx) => models[idx].clone(),
785            }
786        }
787        None => {
788            // Live fetch failed or returned empty (provider doesn't expose
789            // a no-auth listing). The underlying error was traced at debug
790            // level; surface a short provider-named nudge to the user and
791            // fall back to manual entry.
792            ui.note(&format!(
793                "Catalog lookup failed for {provider} — enter a model id manually \
794                 (see the provider's docs for the exact format)."
795            ));
796            let default = if is_set { Some(current.as_str()) } else { None };
797            match ui.string("Model id", default).await? {
798                Answer::Back => return Ok(Nav::Back),
799                Answer::Value(v) => v,
800            }
801        }
802    };
803
804    if new_value != current && !new_value.is_empty() {
805        persist(cfg, &model_path, &new_value).await?;
806    }
807    Ok(Nav::Done)
808}

810async fn channels(cfg: &mut Config, ui: &mut dyn OnboardUi, flags: &Flags) -> Result<Nav> {
811    ui.heading(1, "Channels");
812    match skip_if_configured(
813        cfg,
814        ui,
815        flags,
816        "channels",
817        "Channels",
818        section_has_signal(cfg, "channels"),
819    )
820    .await?
821    {
822        SkipNav::Skip => return Ok(Nav::Done),
823        SkipNav::Back => return Ok(Nav::Back),
824        SkipNav::Enter => {}
825    }
826    loop {
827        // Master list of all channels that exist in the schema. Probe on a
828        // clone: init_defaults(Some("channels")) forces every Option<T>
829        // subsection to Some(default), then prop_fields reveals the full
830        // set. Feature-gated channels drop out automatically.
831        let all_channels: Vec<String> = {
832            let mut probe = cfg.clone();
833            probe.init_defaults(Some("channels"));
834            probe
835                .prop_fields()
836                .iter()
837                .filter_map(|f| f.name.strip_prefix("channels."))
838                .filter_map(|suffix| suffix.split_once('.').map(|(head, _)| head.to_string()))
839                .collect::<std::collections::BTreeSet<_>>()
840                .into_iter()
841                .collect()
842        };
843        let configured: std::collections::BTreeSet<String> = cfg
844            .prop_fields()
845            .iter()
846            .filter_map(|f| f.name.strip_prefix("channels."))
847            .filter_map(|suffix| suffix.split_once('.').map(|(head, _)| head.to_string()))
848            .collect();
849
850        let mut options: Vec<SelectItem> = all_channels
851            .iter()
852            .map(|name| {
853                if configured.contains(name) {
854                    SelectItem::with_badge(name.clone(), "[configured]")
855                } else {
856                    SelectItem::new(name.clone())
857                }
858            })
859            .collect();
860        let done_idx = options.len();
861        options.push(SelectItem::new("Done"));
862
863        let idx = match ui.select("Channel", &options, Some(done_idx)).await? {
864            Answer::Back => return Ok(Nav::Back),
865            Answer::Value(i) => i,
866        };
867        if idx == done_idx {
868            break;
869        }
870
871        let picked = &all_channels[idx];
872        let prefix = format!("channels.{picked}");
873        cfg.init_defaults(Some(&prefix));
874        cfg.save().await?;
875        ui.heading(2, picked);
876        // Back inside a channel's subfields bounces to the channel list
877        // (not to the previous section) — user is still inside Channels.
878        let _ = prompt_fields_under(cfg, ui, &prefix, &[], &[]).await?;
879    }
880    mark_completed(cfg, "channels").await?;
881    Ok(Nav::Done)
882}

884async fn memory(cfg: &mut Config, ui: &mut dyn OnboardUi, flags: &Flags) -> Result<Nav> {
885    ui.heading(1, "Memory");
886    if flags.memory.is_none() {
887        match skip_if_configured(
888            cfg,
889            ui,
890            flags,
891            "memory",
892            "Memory",
893            section_has_signal(cfg, "memory"),
894        )
895        .await?
896        {
897            SkipNav::Skip => return Ok(Nav::Done),
898            SkipNav::Back => return Ok(Nav::Back),
899            SkipNav::Enter => {}
900        }
901    }

906        None => {

914        }

917        persist(cfg, "memory.backend", &new_backend).await?;
918    }
919
920    // Back on auto-save bounces to the backend picker (consumed).
921    let _ = prompt_field(cfg, ui, "memory.auto-save", None).await?;
922    mark_completed(cfg, "memory").await?;
923    Ok(Nav::Done)
924}

961async fn tunnel(cfg: &mut Config, ui: &mut dyn OnboardUi, flags: &Flags) -> Result<Nav> {
962    ui.heading(1, "Tunnel");
963    match skip_if_configured(
964        cfg,
965        ui,
966        flags,
967        "tunnel",
968        "Tunnel",
969        section_has_signal(cfg, "tunnel"),
970    )
971    .await?
972    {
973        SkipNav::Skip => return Ok(Nav::Done),
974        SkipNav::Back => return Ok(Nav::Back),
975        SkipNav::Enter => {}
976    }
977
978    loop {
979        // Provider list derived from the schema: each `tunnel.<name>.*` field
980        // in prop_fields() names a real provider. "none" is always valid and
981        // has no sub-config, so it's prepended.
982        let mut provider_names: Vec<String> = cfg
983            .prop_fields()
984            .iter()
985            .filter_map(|f| f.name.strip_prefix("tunnel."))
986            .filter_map(|suffix| suffix.split_once('.').map(|(head, _)| head.to_string()))
987            .collect::<std::collections::BTreeSet<_>>()
988            .into_iter()
989            .collect();
990        provider_names.insert(0, "none".to_string());
991
992        let options: Vec<SelectItem> = provider_names.iter().map(SelectItem::new).collect();
993        let current_provider = cfg.tunnel.provider.clone();
994        let current_idx = provider_names.iter().position(|p| p == &current_provider);
995        let idx = match ui
996            .select("Public tunnel provider", &options, current_idx)
997            .await?
998        {
999            Answer::Back => return Ok(Nav::Back),
1000            Answer::Value(i) => i,
1001        };
1002        let new_provider = provider_names[idx].clone();
1003
1004        if new_provider != current_provider {
1005            persist(cfg, "tunnel.provider", &new_provider).await?;
1006        }
1007
1008        if new_provider == "none" {
1009            break;
1010        }
1011
1012        let prefix = format!("tunnel.{new_provider}");
1013        cfg.init_defaults(Some(&prefix));
1014        cfg.save().await?;
1015        ui.heading(2, &new_provider);
1016        match prompt_fields_under(cfg, ui, &prefix, &[], &[]).await? {
1017            Nav::Back => continue,
1018            Nav::Done => break,
1019        }
1020    }
1021    mark_completed(cfg, "tunnel").await?;
1022    Ok(Nav::Done)
1023}
