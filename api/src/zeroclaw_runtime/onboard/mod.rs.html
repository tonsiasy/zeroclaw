1//! Onboard orchestrator.
2//!
3//! Thin dispatcher above the `OnboardUi` trait (defined in
4//! `zeroclaw-config::traits`). Section-scoped entry points let callers run
5//! just one slice (`zeroclaw onboard channels`) or the whole flow.
6//!
7//! Everything writes through `Config::set_prop` (or its helpers); direct
8//! struct-field assignment is off-limits per the DRY contract (#5951).

20/// Internal prompt / section navigation signal. `Done` = advance. `Back` =
21/// the user pressed Esc; rewind one step. Helpers propagate it up through
22/// `prompt_field` → `prompt_fields_under` → section fn → `run_all`.
23#[derive(Debug, Clone, Copy, PartialEq, Eq)]
24enum Nav {
25    Done,
26    Back,
27}
28
29/// Skip-gate outcome. `Skip` = section already configured, user chose not
30/// to reconfigure. `Enter` = walk the section. `Back` = user pressed Esc
31/// at the skip prompt, bounce to the previous section.
32#[derive(Debug, Clone, Copy, PartialEq, Eq)]
33enum SkipNav {
34    Skip,
35    Enter,
36    Back,
37}

78/// Top-level onboard dispatcher.
79pub async fn run(
80    cfg: &mut Config,
81    ui: &mut dyn OnboardUi,
82    section: Section,
83    flags: &Flags,
84) -> Result<()> {
85    match section {
86        Section::All => run_all(cfg, ui, flags).await,
87        Section::Workspace => {
88            let _ = workspace(cfg, ui, flags).await?;
89            Ok(())
90        }
91        Section::Providers => {
92            let _ = providers(cfg, ui, flags).await?;
93            Ok(())
94        }
95        Section::Channels => {
96            let _ = channels(cfg, ui, flags).await?;
97            Ok(())
98        }
99        Section::Memory => {
100            let _ = memory(cfg, ui, flags).await?;
101            Ok(())
102        }
103        Section::Hardware => {
104            let _ = hardware(cfg, ui, flags).await?;
105            Ok(())
106        }
107        Section::Tunnel => {
108            let _ = tunnel(cfg, ui, flags).await?;
109            Ok(())
110        }
111    }
112}

114/// Walk every section in order with section-level Back. Each section returns
115/// `Nav::Back` when the user pressed Esc at its first prompt; the loop
116/// rewinds to the previous section. Back at the first section exits
117/// onboarding cleanly (user bails out).
118async fn run_all(cfg: &mut Config, ui: &mut dyn OnboardUi, flags: &Flags) -> Result<()> {
119    let mut i: usize = 0;
120    loop {
121        let nav = match i {
122            0 => workspace(cfg, ui, flags).await?,
123            1 => providers(cfg, ui, flags).await?,
124            2 => channels(cfg, ui, flags).await?,
125            3 => memory(cfg, ui, flags).await?,
126            4 => hardware(cfg, ui, flags).await?,
127            5 => tunnel(cfg, ui, flags).await?,
128            _ => return Ok(()),
129        };
130        match nav {
131            Nav::Done => i += 1,
132            Nav::Back => {
133                if i == 0 {
134                    return Ok(());
135                }
136                i -= 1;
137            }
138        }
139    }
140}

142/// Write a single property and immediately persist the whole config. This is
143/// the ONE path every section takes to mutate cfg, so users who Ctrl+C
144/// mid-flow find their prior answers already saved on disk — re-running
145/// `zeroclaw onboard` picks up where they left off.
146async fn persist(cfg: &mut Config, path: &str, value: &str) -> Result<()> {
147    cfg.set_prop(path, value)?;
148    cfg.save().await?;
149    Ok(())
150}

154/// Per-field default override. When a section knows a sensible default
155/// that lives outside the config (e.g. `AnthropicProvider::default_temperature()`),
156/// it builds a list of these and passes them to `prompt_fields_under`.
157/// The prompt surfaces the default — shown in the label as
158/// `"timeout-secs (default: 120)"` and prefilled into the input so the
159/// user just hits Enter to accept — only when the field is unset in cfg.
160#[derive(Debug, Clone)]
161pub struct FieldDefault {
162    pub path: String,
163    pub display: String,
164}

166fn find_default<'a>(defaults: &'a [FieldDefault], path: &str) -> Option<&'a str> {
167    defaults
168        .iter()
169        .find(|d| d.path == path)
170        .map(|d| d.display.as_str())
171}

173/// True when `input` parses as the same `Vec<String>` form `config.toml`
174/// emits. Lets the StringArray prompt accept the bracketed display form
175/// bidirectionally.
176fn parses_as_string_array(input: &str) -> bool {
177    toml::from_str::<std::collections::HashMap<String, Vec<String>>>(&format!("v = {input}"))
178        .is_ok()

181/// Prompt for a single config field identified by its dotted name. Returns
182/// `Nav::Back` when the user pressed Esc at the prompt; `Nav::Done` on any
183/// other outcome (including "kept current value"). `default` is the
184/// section-supplied fallback for unset fields — surfaced in the label and
185/// prefilled into the input.
186async fn prompt_field(
187    cfg: &mut Config,
188    ui: &mut dyn OnboardUi,
189    name: &str,
190    default: Option<&str>,
191) -> Result<Nav> {
192    let field = cfg
193        .prop_fields()
194        .into_iter()
195        .find(|f| f.name == name)
196        .ok_or_else(|| anyhow::anyhow!("unknown config field: {name}"))?;
197
198    let short = name.rsplit('.').next().unwrap_or(name);
199    let current = field.display_value;
200    // For bools, `display_value` is always `"true"` or `"false"` — never
201    // empty, never `"<unset>"` — so a naive is-set check can't tell an
202    // explicit user choice apart from the struct-level default. Treat
203    // bools as unset here: the [Yes]/[No] toggle already surfaces the
204    // current state, and collapsing `is_set` lets any passed `default`
205    // render in the prompt label (`enabled (default: true)`) while
206    // keeping the misleading "Current: …" annotation out of the help.
207    let is_set = field.kind != PropKind::Bool && !current.is_empty() && current != "<unset>";
208
209    // Surface the docstring as help text above the prompt, and append
210    // whichever annotation fits the prompt's state: "Default: X" when
211    // the section supplied one and the field is unset, "Current: X"
212    // when the config carries a user-set value (non-bool only).
213    let mut help = field.description.to_string();
214    // List-of-strings fields take comma-separated input. Without this
215    // hint users guess and end up entering things like `["alice"]` as
216    // raw text — the parser then treats that as one big string element
217    // and the saved config is garbage.
218    if field.kind == PropKind::StringArray {
219        if !help.is_empty() {
220            help.push('\n');
221        }
222        help.push_str("Format: alice,bob or [\"alice\", \"bob\"]. Empty = clear list.");
223    }
224    if !is_set
225        && let Some(d) = default
226        && !d.is_empty()
227    {
228        if !help.is_empty() {
229            help.push('\n');
230        }
231        help.push_str(&format!("Default: {d}. Press Enter to accept."));
232    } else if is_set {
233        if !help.is_empty() {
234            help.push('\n');
235        }
236        help.push_str(&format!("Current: {current}. Enter to keep."));
237    }
238    ui.note(&help);
239
240    // Label decorates the short name with the default (when visible) so the
241    // value is anchored to the prompt line itself, not just the help text.
242    let prompt_label = match (is_set, default) {
243        (false, Some(d)) if !d.is_empty() => format!("{short} (default: {d})"),
244        _ => short.to_string(),
245    };
246    let prompt = prompt_label.as_str();
247
248    if field.is_secret {
249        match ui.secret(prompt, is_set).await? {
250            Answer::Back => return Ok(Nav::Back),
251            Answer::Value(Some(value)) => persist(cfg, name, &value).await?,
252            Answer::Value(None) => {}
253        }
254        return Ok(Nav::Done);
255    }
256
257    match field.kind {
258        PropKind::Bool => {
259            let cur = current.parse::<bool>().unwrap_or(false);
260            match ui.confirm(prompt, cur).await? {
261                Answer::Back => return Ok(Nav::Back),
262                Answer::Value(new) if new != cur => persist(cfg, name, &new.to_string()).await?,
263                Answer::Value(_) => {}
264            }
265        }
266        PropKind::String | PropKind::Integer | PropKind::Float => {
267            // Prefill priority: config current value > section default > empty.
268            // When the user accepts the prefilled default (no edit), we
269            // still write it through set_prop so the config records the
270            // resolved value rather than leaving it as an implicit fallback.
271            let prefill = if is_set {
272                Some(current.as_str())
273            } else {
274                default
275            };
276            match ui.string(prompt, prefill).await? {
277                Answer::Back => return Ok(Nav::Back),
278                Answer::Value(new) => {
279                    if (is_set || !new.is_empty()) && new != current {
280                        persist(cfg, name, &new).await?;
281                    }
282                }
283            }
284        }
285        PropKind::StringArray => {
286            let prefill = if is_set {
287                Some(current.as_str())
288            } else {
289                default
290            };
291            // Accepts comma-separated input or the bracketed form from
292            // config.toml. Reject malformed brackets — otherwise the
293            // parser silently coerces them into a single-element list
294            // of garbage.
295            loop {
296                match ui.string(prompt, prefill).await? {
297                    Answer::Back => return Ok(Nav::Back),
298                    Answer::Value(new) => {
299                        let trimmed = new.trim();
300                        if trimmed.starts_with('[') && !parses_as_string_array(trimmed) {
301                            ui.note("Invalid array. Use alice,bob or [\"alice\", \"bob\"].");
302                            continue;
303                        }
304                        if (is_set || !new.is_empty()) && new != current {
305                            persist(cfg, name, &new).await?;
306                        }
307                        ui.note("");
308                        break;
309                    }
310                }
311            }
312        }
313        PropKind::Enum => {
314            let variants = field.enum_variants.map(|get| get()).unwrap_or_default();
315            if variants.is_empty() {
316                ui.warn(&format!("skipping {name}: no enum variants exposed"));
317                return Ok(Nav::Done);
318            }
319            let items: Vec<SelectItem> = variants.iter().map(SelectItem::new).collect();
320            let current_idx = if is_set {
321                variants.iter().position(|v| v == &current)
322            } else {
323                default.and_then(|d| variants.iter().position(|v| v == d))
324            };
325            match ui.select(prompt, &items, current_idx).await? {
326                Answer::Back => return Ok(Nav::Back),
327                Answer::Value(idx) => {
328                    let new = &variants[idx];
329                    if new != &current {
330                        persist(cfg, name, new).await?;
331                    }
332                }
333            }
334        }
335    }
336    Ok(Nav::Done)
337}

339/// Iterate every field under `prefix` in `prop_fields()` and prompt for each.
340/// `excludes` lists leaf field names to skip. `defaults` carries per-field
341/// fallback values (e.g. provider-trait defaults) surfaced in the prompt
342/// when the field is unset. Rewinds on `Nav::Back`; propagates `Back` to
343/// the caller when the user rewinds past the first prompt.
344async fn prompt_fields_under(
345    cfg: &mut Config,
346    ui: &mut dyn OnboardUi,
347    prefix: &str,
348    excludes: &[&str],
349    defaults: &[FieldDefault],
350) -> Result<Nav> {
351    let names: Vec<String> = cfg
352        .prop_fields()
353        .into_iter()
354        .filter_map(|f| {
355            let suffix = f.name.strip_prefix(prefix)?.strip_prefix('.')?;
356            if suffix.contains('.') || excludes.contains(&suffix) {
357                return None;
358            }
359            Some(f.name.to_string())
360        })
361        .collect();
362    let mut i: usize = 0;
363    while i < names.len() {
364        let default = find_default(defaults, &names[i]);
365        match prompt_field(cfg, ui, &names[i], default).await? {
366            Nav::Done => i += 1,
367            Nav::Back => {
368                if i == 0 {
369                    return Ok(Nav::Back);
370                }
371                i -= 1;
372            }
373        }
374    }
375    Ok(Nav::Done)
376}

378/// Section-level skip gate. A section is "already configured" when EITHER
379/// (a) it has a marker in `onboard_state.completed_sections` (user finished
380/// the flow once), OR (b) the caller supplies a section-specific
381/// has-meaningful-config signal (e.g. workspace.enabled == true, providers
382/// has a fallback + api-key set). `--force` bypasses unconditionally.
383async fn skip_if_configured(
384    cfg: &Config,
385    ui: &mut dyn OnboardUi,
386    flags: &Flags,
387    section_key: &str,
388    label: &str,
389    has_signal: bool,
390) -> Result<SkipNav> {
391    if flags.force {
392        return Ok(SkipNav::Enter);
393    }
394    let seen = cfg
395        .onboard_state
396        .completed_sections
397        .iter()
398        .any(|s| s == section_key);
399    if !seen && !has_signal {
400        return Ok(SkipNav::Enter);
401    }
402    match ui
403        .confirm(
404            &format!("{label} is already configured. Reconfigure?"),
405            false,
406        )
407        .await?
408    {
409        Answer::Back => Ok(SkipNav::Back),
410        Answer::Value(true) => Ok(SkipNav::Enter),
411        Answer::Value(false) => Ok(SkipNav::Skip),
412    }
413}

415/// Per-section meaningful-config detector used as the secondary skip-gate
416/// signal alongside the completed_sections marker. Returns true when the
417/// section has values that can only come from user action (i.e. diverged
418/// from `Config::default()`'s idle state).
419fn section_has_signal(cfg: &Config, section_key: &str) -> bool {
420    match section_key {
421        "workspace" => cfg.workspace.enabled,
422        "providers" => cfg.providers.fallback.is_some() && !cfg.providers.models.is_empty(),
423        // `channels.cli: bool` is a default-true scalar that lives directly
424        // under `channels.*`, so a bare `starts_with("channels.")` check
425        // fires on every fresh install. Require a nested channel config
426        // (e.g. `channels.telegram.bot-token`) — anything with a second dot
427        // segment — to count as user-driven signal.
428        "channels" => cfg.prop_fields().iter().any(|f| {
429            f.name
430                .strip_prefix("channels.")
431                .is_some_and(|rest| rest.contains('.'))
432        }),
433        "hardware" => cfg.hardware.enabled,
434        // Memory's default backend is "sqlite" and Tunnel's is "none" — both
435        // are valid user choices indistinguishable from untouched defaults.
436        // Marker-only for these two.
437        _ => false,
438    }
439}

452fn openai_compat_models_endpoint(base_url: &str) -> Result<reqwest::Url> {
453    let raw = base_url.trim();
454    if raw.is_empty() {
455        anyhow::bail!("OpenAI-compatible model discovery requires a base URL");
456    }
457
458    let mut endpoint = reqwest::Url::parse(raw)
459        .with_context(|| format!("OpenAI-compatible base URL is invalid: {raw}"))?;
460    if !matches!(endpoint.scheme(), "http" | "https") {
461        anyhow::bail!("OpenAI-compatible base URL must use http:// or https://");
462    }
463
464    let path = endpoint.path().trim_end_matches('/');
465    if path.ends_with("/models") {
466        endpoint.set_query(None);
467        endpoint.set_fragment(None);
468        return Ok(endpoint);
469    }
470
471    let suffix = if path.ends_with("/v1") || path.contains("/v1/") {
472        "models"
473    } else {
474        "v1/models"
475    };
476    let next_path = if path.is_empty() {
477        format!("/{suffix}")
478    } else {
479        format!("{path}/{suffix}")
480    };
481    endpoint.set_path(&next_path);
482    endpoint.set_query(None);
483    endpoint.set_fragment(None);
484    Ok(endpoint)
485}

495async fn discover_openai_compat_models_with_timeout(
496    base_url: &str,
497    api_key: Option<&str>,
498    timeout: Duration,
499) -> Result<Vec<String>> {
500    let endpoint = openai_compat_models_endpoint(base_url)?;
501    let client = reqwest::Client::builder()
502        .timeout(timeout)
503        .build()
504        .context("failed to build OpenAI-compatible discovery client")?;
505
506    let mut request = client.get(endpoint.clone());
507    if let Some(key) = api_key.map(str::trim).filter(|key| !key.is_empty()) {
508        request = request.bearer_auth(key);
509    }
510
511    let response = request
512        .send()
513        .await
514        .with_context(|| format!("OpenAI-compatible model discovery request failed: {endpoint}"))?;
515    let status = response.status();
516    if !status.is_success() {
517        anyhow::bail!("OpenAI-compatible model discovery failed at {endpoint}: HTTP {status}");
518    }
519
520    let payload: OpenAiModelsResponse = response.json().await.with_context(|| {
521        format!("OpenAI-compatible model discovery returned invalid JSON: {endpoint}")
522    })?;
523    let models: Vec<String> = payload
524        .data
525        .into_iter()
526        .map(|model| model.id.trim().to_string())
527        .filter(|id| !id.is_empty())
528        .collect();
529    if models.is_empty() {
530        anyhow::bail!("OpenAI-compatible model discovery returned no model ids: {endpoint}");
531    }
532    Ok(models)
533}

585// ── Sections ─────────────────────────────────────────────────────────────
586// Each section returns `Nav::Back` when the user hits Esc at the very first
587// prompt. Back from a later prompt within the section rewinds locally (via
588// prompt_fields_under / per-section loop), never propagates to the parent.
589
590async fn workspace(cfg: &mut Config, ui: &mut dyn OnboardUi, flags: &Flags) -> Result<Nav> {
591    ui.heading(1, "Workspace");
592    ui.status(&format!(
593        "Workspace directory: {}",
594        cfg.workspace_dir.display()
595    ));
596    match skip_if_configured(
597        cfg,
598        ui,
599        flags,
600        "workspace",
601        "Workspace",
602        section_has_signal(cfg, "workspace"),
603    )
604    .await?
605    {
606        SkipNav::Skip => return Ok(Nav::Done),
607        SkipNav::Back => return Ok(Nav::Back),
608        SkipNav::Enter => {}
609    }
610
611    loop {
612        match prompt_field(cfg, ui, "workspace.enabled", None).await? {
613            Nav::Back => return Ok(Nav::Back),
614            Nav::Done => {}
615        }
616        if cfg.workspace.enabled {
617            match prompt_fields_under(cfg, ui, "workspace", &["enabled"], &[]).await? {
618                Nav::Back => continue,
619                Nav::Done => break,
620            }
621        } else {
622            break;
623        }
624    }
625
626    mark_completed(cfg, "workspace").await?;
627    Ok(Nav::Done)
628}

630async fn providers(cfg: &mut Config, ui: &mut dyn OnboardUi, flags: &Flags) -> Result<Nav> {
631    ui.heading(1, "Providers");
632    if flags.provider.is_none() && flags.api_key.is_none() && flags.model.is_none() {
633        match skip_if_configured(
634            cfg,
635            ui,
636            flags,
637            "providers",
638            "Providers",
639            section_has_signal(cfg, "providers"),
640        )
641        .await?
642        {
643            SkipNav::Skip => return Ok(Nav::Done),
644            SkipNav::Back => return Ok(Nav::Back),
645            SkipNav::Enter => {}
646        }
647    }
648    // Surface both auth paths up front so users with an existing key go
649    // straight to the api_key prompt, and users on OAuth-only providers
650    // (Codex, Claude Code, etc.) know to use the separate login flow.
651    ui.note(
652        "Paste an API key (e.g. `sk-ant-…` for Anthropic, `sk-…` for OpenAI) \
653         when prompted. For OAuth-based providers run: \
654         zeroclaw auth login --provider <name>",
655    );
656
657    // Menu is driven by zeroclaw_providers::list_providers() — single source
658    // of truth for canonical names, display names, aliases.
659    let entries = zeroclaw_providers::list_providers();
660
661    loop {
662        let current_fallback = cfg.providers.fallback.clone().unwrap_or_default();
663
664        let (picked, selected_base_url) = match &flags.provider {
665            Some(forced) => (forced.clone(), None),
666            None => {

668                let mut options: Vec<SelectItem> = entries
669                    .iter()
670                    .map(|p| {
671                        let configured = cfg.providers.models.contains_key(p.name);
672                        let is_active = p.name == current_fallback;
673                        let badge = match (is_active, configured) {
674                            (true, _) => Some("[active]".into()),
675                            (_, true) => Some("[configured]".into()),
676                            _ => None,
677                        };
678                        SelectItem {
679                            label: p.display_name.to_string(),
680                            badge,
681                        }
682                    })
683                    .collect();
684                let custom_idx = options.len();
685                options.push(SelectItem::new(CUSTOM_OPENAI_COMPAT_LABEL));
686                // "Done" lets the user exit providers without picking one —
687                // matches the channels picker's escape hatch. Highlight it
688                // by default when no fallback is set yet (first-time setup).
689                let done_idx = options.len();
690                options.push(SelectItem::new("Done"));
691                let initial = current_idx.or(Some(done_idx));
692                let idx = match ui.select("Provider", &options, initial).await? {
693                    Answer::Back => return Ok(Nav::Back),
694                    Answer::Value(idx) => idx,
695                };
696                if idx == done_idx {
697                    break;
698                }
699                if idx == custom_idx {
700                    let Some(base_url) = prompt_custom_openai_base_url(ui).await? else {
701                        continue;
702                    };
703                    (format!("custom:{base_url}"), Some(base_url))
704                } else {
705                    (entries[idx].name.to_string(), None)
706                }
707            }
708        };
709
710        // Seed the HashMap entry in memory so `prop_fields` can enumerate
711        // its fields for the prompts below. Not persisted here — the first
712        // `persist()` for a real value (api_key, model, …) carries it to
713        // disk. If the user backs out before any value is set, the back
714        // paths drop the entry so it never reaches the file.
715        let is_new_entry = !cfg.providers.models.contains_key(&picked);
716        cfg.providers.models.entry(picked.clone()).or_default();
717
718        // For fresh entries, pre-populate the provider's trait-level defaults
719        // into the in-memory entry. Skipped when reconfiguring so existing
720        // user overrides aren't clobbered. Lives in memory until the first
721        // `persist()` carries the entry — defaults included — to disk.
722        if is_new_entry {
723            let prefix = format!("providers.models.{picked}");
724            for trait_default in provider_trait_defaults(&picked, &prefix) {
725                cfg.set_prop(&trait_default.path, &trait_default.display)?;
726            }
727        }
728        if let Some(base_url) = selected_base_url.as_deref() {
729            cfg.set_prop(&format!("providers.models.{picked}.base-url"), base_url)?;
730        }
731
732        let display_name = entries
733            .iter()
734            .find(|p| p.name == picked)
735            .map(|p| p.display_name)
736            .unwrap_or_else(|| {
737                if picked.starts_with("custom:") {
738                    CUSTOM_OPENAI_COMPAT_LABEL
739                } else {
740                    picked.as_str()
741                }
742            });
743        ui.heading(2, display_name);
744
745        // Apply CLI-flag overrides up front, then skip those names in the
746        // interactive pass so the user isn't re-prompted for what they already
747        // passed on the command line.
748        let prefix = format!("providers.models.{picked}");
749        let api_key_path = format!("{prefix}.api-key");
750        if let Some(api_key) = &flags.api_key {
751            persist(cfg, &api_key_path, api_key).await?;
752        }
753        if let Some(model) = &flags.model {
754            persist(cfg, &format!("{prefix}.model"), model).await?;
755        }
756
757        // Authentication phase is prompted explicitly so the user sees a
758        // clear "API key" step, not a generic `api-key (stored, replace?)`
759        // lost among other fields. The heading(2) also overrides the
760        // provider subsection so the panel reads "Providers › Authentication".
761        if flags.api_key.is_none() {
762            ui.heading(2, &format!("{display_name} › Authentication"));
763            match prompt_field(cfg, ui, &api_key_path, None).await? {
764                Nav::Back => {
765                    if flags.provider.is_some() {
766                        return Ok(Nav::Back);
767                    }
768                    cfg.providers.models.remove(&picked);
769                    continue;
770                }
771                Nav::Done => {}
772            }
773            ui.heading(2, display_name);
774        }
775
776        if flags.model.is_none() {
777            ui.heading(2, &format!("{display_name} › Model"));
778            match prompt_model(cfg, ui, &picked).await? {
779                Nav::Back => {
780                    if flags.provider.is_some() {
781                        return Ok(Nav::Back);
782                    }
783                    cfg.providers.models.remove(&picked);
784                    continue;
785                }
786                Nav::Done => {}
787            }
788            ui.heading(2, display_name);
789        }
790
791        // Advanced settings (temperature, timeout, base-url override,
792        // wire-api, etc.) are gated behind an opt-in. Most users never
793        // touch these, and the trait-level defaults are sensible.
794        match offer_advanced_settings(cfg, ui, &picked, &prefix).await? {
795            Nav::Back => {
796                if flags.provider.is_some() {
797                    return Ok(Nav::Back);
798                }
799                continue;
800            }
801            Nav::Done => {}
802        }
803
804        if cfg.providers.fallback.as_deref() != Some(picked.as_str()) {
805            persist(cfg, "providers.fallback", &picked).await?;
806        }
807
808        break;
809    }
810
811    mark_completed(cfg, "providers").await?;
812    Ok(Nav::Done)
813}

820    cfg: &mut Config,
821    ui: &mut dyn OnboardUi,
822    provider: &str,
823    prefix: &str,

847/// Build the `FieldDefault` list for a provider from its trait methods.
848/// Single source of truth: used to pre-populate fresh entries during
849/// onboarding AND to surface defaults in the advanced-settings walk.
850/// A provider handle constructed with `api_key = None` can still answer
851/// the `default_*` questions — they're family-level constants, not
852/// runtime values. If construction fails (unknown / local-only provider),
853/// returns an empty vec and callers fall back to per-field display values.
854fn provider_trait_defaults(provider: &str, prefix: &str) -> Vec<FieldDefault> {
855    let Ok(handle) = zeroclaw_providers::create_provider(provider, None) else {
856        return Vec::new();
857    };
858    let mut v = vec![
859        FieldDefault {
860            path: format!("{prefix}.temperature"),
861            display: handle.default_temperature().to_string(),
862        },
863        FieldDefault {
864            path: format!("{prefix}.max-tokens"),
865            display: handle.default_max_tokens().to_string(),
866        },
867        FieldDefault {
868            path: format!("{prefix}.timeout-secs"),
869            display: handle.default_timeout_secs().to_string(),
870        },
871        FieldDefault {
872            path: format!("{prefix}.wire-api"),
873            display: handle.default_wire_api().to_string(),
874        },
875    ];
876    if let Some(url) = handle.default_base_url() {
877        v.push(FieldDefault {
878            path: format!("{prefix}.base-url"),
879            display: url.to_string(),
880        });
881    }
882    v
883}

885/// Exclude fields that don't apply to the selected provider family.
886///
887/// A field absent from this filter is shown for every provider. Grow the
888/// list as provider-specific fields land on `ModelProviderConfig` — the
889/// goal is to keep the advanced walk focused on fields the user might
890/// reasonably want to override for the provider they picked.

892/// Follow-up (#5960 deferral): a `#[configurable(family = "...")]`

896fn provider_family_excludes(provider: &str) -> Vec<&'static str> {

899    if provider != "azure_openai" {
900        out.push("azure-openai-resource");
901        out.push("azure-openai-deployment");
902        out.push("azure-openai-api-version");

905    if !matches!(provider, "openai" | "openai_codex") {
906        out.push("wire-api");
907        out.push("requires-openai-auth");
908    }
909    out
910}

912/// Prompt for the model field using the provider's live model catalog.
913///
914/// Calls `Provider::list_models()` (no auth — see `zeroclaw-providers`
915/// models_dev + native public endpoints). Falls back to a manual string
916/// input when the provider doesn't expose a no-auth list or the fetch fails.
917async fn prompt_model(cfg: &mut Config, ui: &mut dyn OnboardUi, provider: &str) -> Result<Nav> {
918    let model_path = format!("providers.models.{provider}.model");
919    let current = cfg.get_prop(&model_path).unwrap_or_default();
920    let is_set = !current.is_empty() && current != "<unset>";
921    let profile = cfg.providers.models.get(provider);
922    let api_key = profile.and_then(|entry| entry.api_key.as_deref());
923    let configured_base_url = profile.and_then(|entry| entry.base_url.as_deref());
924    let discovery_base_url = openai_compat_discovery_base_url(provider, configured_base_url);
925    let should_try_openai_compat =
926        provider.trim().starts_with("custom:") || !is_known_provider_name(provider);
927
928    let catalog_models = match zeroclaw_providers::create_provider(provider, None) {
929        Ok(handle) => {
930            ui.status("Fetching models...");
931            match handle.list_models().await {
932                Ok(models) => Some(models),
933                Err(e) => {
934                    tracing::debug!(provider, error = ?e, "models.dev catalog fetch failed");
935                    None
936                }
937            }
938        }
939        Err(e) => {
940            tracing::debug!(
941                provider,
942                error = ?e,
943                "provider construction failed for model-list probe"
944            );
945            None
946        }
947    };
948    let live_models = match catalog_models.filter(|ms| !ms.is_empty()) {
949        Some(models) => Some(models),
950        None if should_try_openai_compat => {
951            if let Some(base_url) = discovery_base_url.as_deref() {
952                ui.status("Fetching models from /v1/models...");
953                match discover_openai_compat_models(base_url, api_key).await {
954                    Ok(models) => Some(models),
955                    Err(e) => {
956                        tracing::debug!(
957                            provider,
958                            base_url,
959                            error = ?e,
960                            "OpenAI-compatible model discovery failed"
961                        );
962                        None
963                    }
964                }
965            } else {
966                None
967            }
968        }
969        None => None,
970    };
971
972    let new_value = match live_models {
973        Some(models) => {
974            let items: Vec<SelectItem> = models.iter().map(SelectItem::new).collect();
975            let current_idx = models.iter().position(|m| m == &current);
976            match ui.select("Model", &items, current_idx).await? {
977                Answer::Back => return Ok(Nav::Back),
978                Answer::Value(idx) => models[idx].clone(),
979            }
980        }
981        None => {
982            // Live fetch failed or returned empty (provider doesn't expose
983            // a no-auth listing). The underlying error was traced at debug
984            // level; surface a short provider-named nudge to the user and
985            // fall back to manual entry.
986            ui.note(&format!(
987                "Catalog lookup failed for {provider} — enter a model id manually \
988                 (see the provider's docs for the exact format)."
989            ));
990            let default = if is_set { Some(current.as_str()) } else { None };
991            match ui.string("Model id", default).await? {
992                Answer::Back => return Ok(Nav::Back),
993                Answer::Value(v) => v,
994            }
995        }
996    };
997
998    if new_value != current && !new_value.is_empty() {
999        persist(cfg, &model_path, &new_value).await?;
1000    }
1001    Ok(Nav::Done)
1002}
