8/// Context for cost tracking within the tool call loop.
9/// Scoped via `tokio::task_local!` at call sites (channels, gateway).
10#[derive(Clone)]
11pub struct ToolLoopCostTrackingContext {
12    pub tracker: Arc<CostTracker>,
13    pub prices: Arc<std::collections::HashMap<String, ModelPricing>>,
14}
15
16impl ToolLoopCostTrackingContext {
17    pub fn new(
18        tracker: Arc<CostTracker>,
19        prices: Arc<std::collections::HashMap<String, ModelPricing>>,
20    ) -> Self {
21        Self { tracker, prices }
22    }
23}
24
25tokio::task_local! {
26    pub static TOOL_LOOP_COST_TRACKING_CONTEXT: Option<ToolLoopCostTrackingContext>;
27}

29/// Record token usage from an LLM response via the task-local cost tracker.
30/// Returns `(total_tokens, cost_usd)` on success, `None` when not scoped or no usage.
31pub fn record_tool_loop_cost_usage(
32    provider_name: &str,
33    model: &str,
34    usage: &zeroclaw_providers::traits::TokenUsage,
35) -> Option<(u64, f64)> {
36    let input_tokens = usage.input_tokens.unwrap_or(0);
37    let output_tokens = usage.output_tokens.unwrap_or(0);
38    let total_tokens = input_tokens.saturating_add(output_tokens);
39    if total_tokens == 0 {
40        return None;
41    }
42
43    let ctx = TOOL_LOOP_COST_TRACKING_CONTEXT
44        .try_with(Clone::clone)
45        .ok()
46        .flatten()?;
47    // 3-tier model pricing lookup: direct name → provider/model → suffix after last `/`
48    let pricing = ctx
49        .prices
50        .get(model)
51        .or_else(|| ctx.prices.get(&format!("{provider_name}/{model}")))
52        .or_else(|| {
53            model
54                .rsplit_once('/')
55                .and_then(|(_, suffix)| ctx.prices.get(suffix))
56        });
57    let cost_usage = CostTokenUsage::new(
58        model,
59        input_tokens,
60        output_tokens,
61        pricing.map_or(0.0, |entry| entry.input),
62        pricing.map_or(0.0, |entry| entry.output),
63    );
64
65    if pricing.is_none() {
66        tracing::debug!(
67            provider = provider_name,
68            model,
69            "Cost tracking recorded token usage with zero pricing (no pricing entry found)"
70        );
71    }
72
73    if let Err(error) = ctx.tracker.record_usage(cost_usage.clone()) {
74        tracing::warn!(
75            provider = provider_name,
76            model,
77            "Failed to record cost tracking usage: {error}"
78        );
79    }
80
81    Some((cost_usage.total_tokens, cost_usage.cost_usd))
82}

84/// Check budget before an LLM call. Returns `None` when no cost tracking
85/// context is scoped (tests, delegate, CLI without cost config).
86pub fn check_tool_loop_budget() -> Option<BudgetCheck> {
87    TOOL_LOOP_COST_TRACKING_CONTEXT
88        .try_with(Clone::clone)
89        .ok()
90        .flatten()
91        .map(|ctx| {
92            ctx.tracker
93                .check_budget(0.0)
94                .unwrap_or(BudgetCheck::Allowed)
95        })
96}
