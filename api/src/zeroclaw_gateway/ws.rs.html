1//! WebSocket agent chat handler.
2//!
3//! Connect: `ws://host:port/ws/chat?session_id=ID&name=My+Session`
4//!
5//! Protocol:
6//! ```text
7//! Server -> Client: {"type":"session_start","session_id":"...","name":"...","resumed":true,"message_count":42}
8//! Client -> Server: {"type":"message","content":"Hello"}
9//! Server -> Client: {"type":"chunk","content":"Hi! "}
10//! Server -> Client: {"type":"tool_call","name":"shell","args":{...}}
11//! Server -> Client: {"type":"tool_result","name":"shell","output":"..."}
12//! Server -> Client: {"type":"done","full_response":"..."}
13//! ```
14//!
15//! Query params:
16//! - `session_id` — resume or create a session (default: new UUID)
17//! - `name` — optional human-readable label for the session
18//! - `token` — bearer auth token (alternative to Authorization header)
19

33/// Optional connection parameters sent as the first WebSocket message.
34///
35/// If the first message after upgrade is `{"type":"connect",...}`, these
36/// parameters are extracted and an acknowledgement is sent back. Old clients
37/// that send `{"type":"message",...}` as the first frame still work — the
38/// message is processed normally (backward-compatible).
39#[derive(Debug, Deserialize)]
40struct ConnectParams {
41    #[serde(rename = "type")]
42    msg_type: String,
43    /// Client-chosen session ID for memory persistence
44    #[serde(default)]
45    session_id: Option<String>,
46    /// Device name for device registry tracking
47    #[serde(default)]
48    device_name: Option<String>,
49    /// Client capabilities
50    #[serde(default)]
51    capabilities: Vec<String>,
52}
53
54/// The sub-protocol we support for the chat WebSocket.
55const WS_PROTOCOL: &str = "zeroclaw.v1";
56
57/// Prefix used in `Sec-WebSocket-Protocol` to carry a bearer token.
58const BEARER_SUBPROTO_PREFIX: &str = "bearer.";
59
60#[derive(Deserialize)]
61pub struct WsQuery {
62    pub token: Option<String>,
63    pub session_id: Option<String>,
64    /// Optional human-readable name for the session.
65    pub name: Option<String>,
66}
67
68/// Extract a bearer token from WebSocket-compatible sources.
69///
70/// Precedence (first non-empty wins):
71/// 1. `Authorization: Bearer <token>` header
72/// 2. `Sec-WebSocket-Protocol: bearer.<token>` subprotocol
73/// 3. `?token=<token>` query parameter
74///
75/// Browsers cannot set custom headers on `new WebSocket(url)`, so the query
76/// parameter and subprotocol paths are required for browser-based clients.
77fn extract_ws_token<'a>(headers: &'a HeaderMap, query_token: Option<&'a str>) -> Option<&'a str> {
78    // 1. Authorization header
79    if let Some(t) = headers
80        .get(header::AUTHORIZATION)
81        .and_then(|v| v.to_str().ok())
82        .and_then(|auth| auth.strip_prefix("Bearer "))
83        && !t.is_empty()
84    {
85        return Some(t);
86    }
87
88    // 2. Sec-WebSocket-Protocol: bearer.<token>
89    if let Some(t) = headers
90        .get("sec-websocket-protocol")
91        .and_then(|v| v.to_str().ok())
92        .and_then(|protos| {
93            protos
94                .split(',')
95                .map(|p| p.trim())
96                .find_map(|p| p.strip_prefix(BEARER_SUBPROTO_PREFIX))
97        })
98        && !t.is_empty()
99    {
100        return Some(t);
101    }
102
103    // 3. ?token= query parameter
104    if let Some(t) = query_token
105        && !t.is_empty()
106    {
107        return Some(t);
108    }
109
110    None
111}

113/// GET /ws/chat — WebSocket upgrade for agent chat
114pub async fn handle_ws_chat(
115    State(state): State<AppState>,
116    Query(params): Query<WsQuery>,
117    headers: HeaderMap,
118    ws: WebSocketUpgrade,
119) -> impl IntoResponse {
120    // Auth: check header, subprotocol, then query param (precedence order)
121    if state.pairing.require_pairing() {
122        let token = extract_ws_token(&headers, params.token.as_deref()).unwrap_or("");
123        if !state.pairing.is_authenticated(token) {
124            return (
125                axum::http::StatusCode::UNAUTHORIZED,
126                "Unauthorized — provide Authorization header, Sec-WebSocket-Protocol bearer, or ?token= query param",
127            )
128                .into_response();
129        }
130    }
131
132    // Echo Sec-WebSocket-Protocol if the client requests our sub-protocol.
133    let ws = if headers
134        .get("sec-websocket-protocol")
135        .and_then(|v| v.to_str().ok())
136        .is_some_and(|protos| protos.split(',').any(|p| p.trim() == WS_PROTOCOL))
137    {
138        ws.protocols([WS_PROTOCOL])
139    } else {
140        ws
141    };
142
143    let session_id = params.session_id;
144    let session_name = params.name;
145    ws.on_upgrade(move |socket| handle_socket(socket, state, session_id, session_name))
146        .into_response()
147}

149/// Gateway session key prefix to avoid collisions with channel sessions.
150const GW_SESSION_PREFIX: &str = "gw_";
151
152async fn handle_socket(
153    socket: WebSocket,
154    state: AppState,
155    session_id: Option<String>,
156    session_name: Option<String>,
157) {
158    let (mut sender, mut receiver) = socket.split();
159
160    // Resolve session ID: use provided or generate a new UUID
161    let session_id = session_id.unwrap_or_else(|| uuid::Uuid::new_v4().to_string());
162    let session_key = format!("{GW_SESSION_PREFIX}{session_id}");
163
164    // Build a persistent Agent for this connection so history is maintained across turns.
165    let config = state.config.lock().clone();
166    let mut agent = match zeroclaw_runtime::agent::Agent::from_config(&config).await {
167        Ok(a) => a,
168        Err(e) => {
169            tracing::error!(error = %e, "Agent initialization failed");
170            let err = serde_json::json!({
171                "type": "error",
172                "message": format!("Failed to initialise agent: {e}"),
173                "code": "AGENT_INIT_FAILED"
174            });
175            let _ = sender.send(Message::Text(err.to_string().into())).await;
176            let _ = sender
177                .send(Message::Close(Some(axum::extract::ws::CloseFrame {
178                    code: 1011,
179                    reason: axum::extract::ws::Utf8Bytes::from_static(
180                        "Agent initialization failed",
181                    ),
182                })))
183                .await;
184            return;
185        }
186    };
187    agent.set_memory_session_id(Some(session_id.clone()));
188
189    // Hydrate agent from persisted session (if available)
190    let mut resumed = false;
191    let mut message_count: usize = 0;
192    let mut effective_name: Option<String> = None;
193    if let Some(ref backend) = state.session_backend {
194        let messages = backend.load(&session_key);
195        if !messages.is_empty() {
196            message_count = messages.len();
197            agent.seed_history(&messages);
198            resumed = true;
199        }
200        // Set session name if provided (non-empty) on connect
201        if let Some(ref name) = session_name
202            && !name.is_empty()
203        {
204            let _ = backend.set_session_name(&session_key, name);
205            effective_name = Some(name.clone());
206        }
207        // If no name was provided via query param, load the stored name
208        if effective_name.is_none() {
209            effective_name = backend.get_session_name(&session_key).unwrap_or(None);
210        }
211    }
212
213    // Send session_start message to client
214    let mut session_start = serde_json::json!({
215        "type": "session_start",
216        "session_id": session_id,
217        "resumed": resumed,
218        "message_count": message_count,
219    });
220    if let Some(ref name) = effective_name {
221        session_start["name"] = serde_json::Value::String(name.clone());
222    }
223    let _ = sender
224        .send(Message::Text(session_start.to_string().into()))
225        .await;
226
227    // ── Optional connect handshake ──────────────────────────────────
228    // The first message may be a `{"type":"connect",...}` frame carrying
229    // connection parameters.  If it is, we extract the params, send an
230    // ack, and proceed to the normal message loop.  If the first message
231    // is a regular `{"type":"message",...}` frame, we fall through and
232    // process it immediately (backward-compatible).
233    let mut first_msg_fallback: Option<String> = None;
234
235    if let Some(first) = receiver.next().await {
236        match first {
237            Ok(Message::Text(text)) => {
238                if let Ok(cp) = serde_json::from_str::<ConnectParams>(&text) {
239                    if cp.msg_type == "connect" {
240                        debug!(
241                            session_id = ?cp.session_id,
242                            device_name = ?cp.device_name,
243                            capabilities = ?cp.capabilities,
244                            "WebSocket connect params received"
245                        );
246                        // Override session_id if provided in connect params
247                        if let Some(sid) = &cp.session_id {
248                            agent.set_memory_session_id(Some(sid.clone()));
249                        }
250                        let ack = serde_json::json!({
251                            "type": "connected",
252                            "message": "Connection established"
253                        });
254                        let _ = sender.send(Message::Text(ack.to_string().into())).await;
255                    } else {
256                        // Not a connect message — fall through to normal processing
257                        first_msg_fallback = Some(text.to_string());
258                    }
259                } else {
260                    // Not parseable as ConnectParams — fall through
261                    first_msg_fallback = Some(text.to_string());
262                }
263            }
264            Ok(Message::Close(_)) | Err(_) => return,
265            _ => {}
266        }
267    }
268
269    // Process the first message if it was not a connect frame
270    if let Some(ref text) = first_msg_fallback {
271        if let Ok(parsed) = serde_json::from_str::<serde_json::Value>(text) {
272            if parsed["type"].as_str() == Some("message") {
273                let content = parsed["content"].as_str().unwrap_or("").to_string();
274                if !content.is_empty() {
275                    // Persist user message
276                    if let Some(ref backend) = state.session_backend {
277                        let user_msg = zeroclaw_providers::ChatMessage::user(&content);
278                        let _ = backend.append(&session_key, &user_msg);
279                    }
280                    process_chat_message(&state, &mut agent, &mut sender, &content, &session_key)
281                        .await;
282                }
283            } else {
284                let unknown_type = parsed["type"].as_str().unwrap_or("unknown");
285                let err = serde_json::json!({
286                    "type": "error",
287                    "message": format!(
288                        "Unsupported message type \"{unknown_type}\". Send {{\"type\":\"message\",\"content\":\"your text\"}}"
289                    )
290                });
291                let _ = sender.send(Message::Text(err.to_string().into())).await;
292            }
293        } else {
294            let err = serde_json::json!({
295                "type": "error",
296                "message": "Invalid JSON. Send {\"type\":\"message\",\"content\":\"your text\"}"
297            });
298            let _ = sender.send(Message::Text(err.to_string().into())).await;
299        }
300    }
301
302    // Subscribe to the shared broadcast channel so cron/heartbeat events
303    // are forwarded to this WebSocket client.
304    let mut broadcast_rx = state.event_tx.subscribe();
305
306    loop {
307        tokio::select! {
308            // ── Client message ────────────────────────────────────────
309            client_msg = receiver.next() => {
310                let Some(msg) = client_msg else { break };
311                let msg = match msg {
312                    Ok(Message::Text(text)) => text,
313                    Ok(Message::Close(_)) | Err(_) => break,
314                    _ => continue,
315                };
316
317                // Parse incoming message
318                let parsed: serde_json::Value = match serde_json::from_str(&msg) {
319                    Ok(v) => v,
320                    Err(e) => {
321                        let err = serde_json::json!({
322                            "type": "error",
323                            "message": format!("Invalid JSON: {}", e),
324                            "code": "INVALID_JSON"
325                        });
326                        let _ = sender.send(Message::Text(err.to_string().into())).await;
327                        continue;
328                    }
329                };
330
331                let msg_type = parsed["type"].as_str().unwrap_or("");
332
333                // ── Voice duplex event dispatch (gated by feature flag + runtime config) ──
334                #[cfg(feature = "gateway-voice-duplex")]
335                {
336                    let duplex_enabled = state
337                        .config
338                        .lock()
339                        .channels
340                        .voice_duplex
341                        .as_ref()
342                        .is_some_and(|v| v.enabled);
343                    if duplex_enabled {
344                        if let Some(voice_event) = crate::voice_duplex::try_parse_voice_event(&msg) {
345                            if let Some(error_frame) = crate::voice_duplex::handle_voice_event(voice_event) {
346                                let _ = sender.send(Message::Text(error_frame.to_string().into())).await;
347                            }
348                            continue;
349                        }
350                    }
351                }
352
353                if msg_type != "message" {
354                    let err = serde_json::json!({
355                        "type": "error",
356                        "message": format!(
357                            "Unsupported message type \"{msg_type}\". Send {{\"type\":\"message\",\"content\":\"your text\"}}"
358                        ),
359                        "code": "UNKNOWN_MESSAGE_TYPE"
360                    });
361                    let _ = sender.send(Message::Text(err.to_string().into())).await;
362                    continue;
363                }
364
365                let content = parsed["content"].as_str().unwrap_or("").to_string();
366                if content.is_empty() {
367                    let err = serde_json::json!({
368                        "type": "error",
369                        "message": "Message content cannot be empty",
370                        "code": "EMPTY_CONTENT"
371                    });
372                    let _ = sender.send(Message::Text(err.to_string().into())).await;
373                    continue;
374                }
375
376                // Acquire session lock to serialize concurrent turns
377                let _session_guard = match state.session_queue.acquire(&session_key).await {
378                    Ok(guard) => guard,
379                    Err(e) => {
380                        let err = serde_json::json!({
381                            "type": "error",
382                            "message": e.to_string(),
383                            "code": "SESSION_BUSY"
384                        });
385                        let _ = sender.send(Message::Text(err.to_string().into())).await;
386                        continue;
387                    }
388                };
389
390                // Persist user message
391                if let Some(ref backend) = state.session_backend {
392                    let user_msg = zeroclaw_providers::ChatMessage::user(&content);
393                    let _ = backend.append(&session_key, &user_msg);
394                }
395
396                process_chat_message(&state, &mut agent, &mut sender, &content, &session_key).await;
397            }
398
399            // ── Broadcast event (cron/heartbeat results) ──────────────
400            event = broadcast_rx.recv() => {
401                if let Ok(event) = event {
402                    let _ = sender.send(Message::Text(event.to_string().into())).await;
403                }
404            }
405        }
406    }
407}

409/// Process a single chat message through the agent and send the response.
410///
411/// Uses [`Agent::turn_streamed`] so that intermediate text chunks, tool calls,
412/// and tool results are forwarded to the WebSocket client in real time.
413async fn process_chat_message(
414    state: &AppState,
415    agent: &mut zeroclaw_runtime::agent::Agent,
416    sender: &mut futures_util::stream::SplitSink<WebSocket, Message>,
417    content: &str,
418    session_key: &str,
419) {
420    use zeroclaw_runtime::agent::TurnEvent;
421
422    let provider_label = state
423        .config
424        .lock()
425        .providers
426        .fallback
427        .clone()
428        .unwrap_or_else(|| "unknown".to_string());
429
430    // Broadcast agent_start event
431    let _ = state.event_tx.send(serde_json::json!({
432        "type": "agent_start",
433        "provider": provider_label,
434        "model": state.model,
435    }));
436
437    // Set session state to running
438    let turn_id = uuid::Uuid::new_v4().to_string();
439    if let Some(ref backend) = state.session_backend {
440        let _ = backend.set_session_state(session_key, "running", Some(&turn_id));
441    }
442
443    // ── Cancellation token lifecycle ─────────────────────────────
444    // Create a token before the turn starts so the abort endpoint
445    // can cancel it. Remove it after the turn completes regardless
446    // of outcome (normal, error, or cancelled).
447    let cancel_token = tokio_util::sync::CancellationToken::new();
448    {
449        state
450            .cancel_tokens
451            .lock()
452            .expect("cancel_tokens lock poisoned")
453            .insert(session_key.to_string(), cancel_token.clone());
454    }
455
456    // Channel for streaming turn events from the agent.
457    let (event_tx, mut event_rx) = tokio::sync::mpsc::channel::<TurnEvent>(64);
458
459    // Run the streamed turn concurrently: the agent produces events
460    // while we forward them to the WebSocket below.  We cannot move
461    // `agent` into a spawned task (it is `&mut`), so we use a join
462    // instead — `turn_streamed` writes to the channel and we drain it
463    // from the other branch.
464    let content_owned = content.to_string();
465    let session_key_owned = session_key.to_string();
466    let turn_fut = async {
467        zeroclaw_runtime::agent::loop_::scope_session_key(
468            Some(session_key_owned),
469            agent.turn_streamed(&content_owned, event_tx, Some(cancel_token.clone())),
470        )
471        .await
472    };
473
474    // Drive both futures concurrently: the agent turn produces events
475    // and we relay them over WebSocket. Track streamed chunks so we
476    // can reconstruct partial content on cancellation.
477    //
478    // WHY incremental persistence: If the process crashes during streaming,
479    // the assistant's response is lost — only the user message survives.
480    // We append a placeholder assistant message on the first chunk, then
481    // update_last periodically (every 500ms) so partial content survives.
482    // The final response overwrites this via update_last on completion.
483    let mut accumulated_text = String::new();
484    let mut partial_saved = false;
485    let mut last_partial_save = std::time::Instant::now();
486    let partial_save_interval = std::time::Duration::from_millis(500);
487
488    let forward_fut = async {
489        while let Some(event) = event_rx.recv().await {
490            let ws_msg = match event {
491                TurnEvent::Chunk { ref delta } => {
492                    accumulated_text.push_str(delta);
493
494                    // Incremental persistence: save partial content so it
495                    // survives a crash. First chunk appends, subsequent
496                    // chunks update in-place.
497                    if last_partial_save.elapsed() >= partial_save_interval {
498                        if let Some(ref backend) = state.session_backend {
499                            let partial =
500                                zeroclaw_providers::ChatMessage::assistant(&accumulated_text);
501                            if partial_saved {
502                                let _ = backend.update_last(session_key, &partial);
503                            } else {
504                                let _ = backend.append(session_key, &partial);
505                                partial_saved = true;
506                            }
507                        }
508                        last_partial_save = std::time::Instant::now();
509                    }
510
511                    serde_json::json!({ "type": "chunk", "content": delta })
512                }
513                TurnEvent::Thinking { delta } => {
514                    serde_json::json!({ "type": "thinking", "content": delta })
515                }
516                TurnEvent::ToolCall { id, name, args } => {
517                    serde_json::json!({ "type": "tool_call", "id": id, "name": name, "args": args })
518                }
519                TurnEvent::ToolResult { id, name, output } => {
520                    serde_json::json!({ "type": "tool_result", "id": id, "name": name, "output": output })
521                }
522            };
523            let _ = sender.send(Message::Text(ws_msg.to_string().into())).await;
524        }
525    };
526
527    let (result, ()) = tokio::join!(turn_fut, forward_fut);
528
529    // ── Remove cancel token (turn finished) ──────────────────────
530    {
531        state
532            .cancel_tokens
533            .lock()
534            .expect("cancel_tokens lock poisoned")
535            .remove(session_key);
536    }
537
538    // Check if this turn was cancelled. `turn_streamed` propagates
539    // `ToolLoopCancelled` through anyhow, so we detect it here.
540    let was_cancelled = match &result {
541        Err(e) => zeroclaw_runtime::agent::loop_::is_tool_loop_cancelled(e),
542        Ok(_) => false,
543    };
544
545    if was_cancelled {
546        // Store partial content with interruption marker so the
547        // conversation stays coherent for subsequent turns.
548        let truncated = if accumulated_text.is_empty() {
549            "[interrupted by user]".to_string()
550        } else {
551            format!("{accumulated_text}\n\n[interrupted by user]")
552        };
553
554        if let Some(ref backend) = state.session_backend {
555            let assistant_msg = zeroclaw_providers::ChatMessage::assistant(&truncated);
556            if partial_saved {
557                let _ = backend.update_last(session_key, &assistant_msg);
558            } else {
559                let _ = backend.append(session_key, &assistant_msg);
560            }
561        }
562
563        // Inform the client the turn was aborted
564        let aborted = serde_json::json!({ "type": "aborted" });
565        let _ = sender.send(Message::Text(aborted.to_string().into())).await;
566
567        // Set session state to idle
568        if let Some(ref backend) = state.session_backend {
569            let _ = backend.set_session_state(session_key, "idle", None);
570        }
571
572        // Broadcast agent_end event
573        let _ = state.event_tx.send(serde_json::json!({
574            "type": "agent_end",
575            "provider": provider_label,
576            "model": state.model,
577        }));
578
579        return;
580    }
581
582    match result {
583        Ok(response) => {
584            // Persist final assistant response. If we saved partial content
585            // during streaming, update it in-place; otherwise append fresh.
586            if let Some(ref backend) = state.session_backend {
587                let assistant_msg = zeroclaw_providers::ChatMessage::assistant(&response);
588                if partial_saved {
589                    let _ = backend.update_last(session_key, &assistant_msg);
590                } else {
591                    let _ = backend.append(session_key, &assistant_msg);
592                }
593            }
594
595            // Fire-and-forget memory consolidation so facts from WS sessions
596            // are extracted to long-term memory (Daily + Core categories).
597            if state.auto_save {
598                let mem = state.mem.clone();
599                let provider = state.provider.clone();
600                let model = state.model.clone();
601                let user_msg = content.to_string();
602                let assistant_resp = response.clone();
603                tokio::spawn(async move {
604                    if let Err(e) = zeroclaw_memory::consolidation::consolidate_turn(
605                        provider.as_ref(),
606                        &model,
607                        mem.as_ref(),
608                        &user_msg,
609                        &assistant_resp,
610                    )
611                    .await
612                    {
613                        tracing::debug!("WS memory consolidation skipped: {e}");
614                    }
615                });
616            }
617
618            // Send chunk_reset so the client clears any accumulated draft
619            // before the authoritative done message.
620            let reset = serde_json::json!({ "type": "chunk_reset" });
621            let _ = sender.send(Message::Text(reset.to_string().into())).await;
622
623            let done = serde_json::json!({
624                "type": "done",
625                "full_response": response,
626            });
627            let _ = sender.send(Message::Text(done.to_string().into())).await;
628
629            // Set session state to idle
630            if let Some(ref backend) = state.session_backend {
631                let _ = backend.set_session_state(session_key, "idle", None);
632            }
633
634            // Broadcast agent_end event
635            let _ = state.event_tx.send(serde_json::json!({
636                "type": "agent_end",
637                "provider": provider_label,
638                "model": state.model,
639            }));
640        }
641        Err(e) => {
642            // Set session state to error
643            if let Some(ref backend) = state.session_backend {
644                let _ = backend.set_session_state(session_key, "error", Some(&turn_id));
645            }
646
647            tracing::error!(error = %e, "Agent turn failed");
648            let sanitized = zeroclaw_providers::sanitize_api_error(&e.to_string());
649            let error_code = if sanitized.to_lowercase().contains("api key")
650                || sanitized.to_lowercase().contains("authentication")
651                || sanitized.to_lowercase().contains("unauthorized")
652            {
653                "AUTH_ERROR"
654            } else if sanitized.to_lowercase().contains("provider")
655                || sanitized.to_lowercase().contains("model")
656            {
657                "PROVIDER_ERROR"
658            } else {
659                "AGENT_ERROR"
660            };
661            let err = serde_json::json!({
662                "type": "error",
663                "message": sanitized,
664                "code": error_code,
665            });
666            let _ = sender.send(Message::Text(err.to_string().into())).await;
667
668            // Broadcast error event
669            let _ = state.event_tx.send(serde_json::json!({
670                "type": "error",
671                "component": "ws_chat",
672                "message": sanitized,
673            }));
674        }
675    }
676}
